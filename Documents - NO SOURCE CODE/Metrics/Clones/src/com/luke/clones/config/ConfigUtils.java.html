109     private static ScreenRatioType getScreenRatio(){
110         int deviceOriginalScreenWidth = Gdx.graphics.getWidth();
111         int deviceOriginalScreenHeight = Gdx.graphics.getHeight();
112         
113         float ratio = (float)deviceOriginalScreenHeight/(float)deviceOriginalScreenWidth;
114         System.out.println("ratio: "+ratio);
115         if(Math.abs(1.777-ratio)<0.01){
116             return ScreenRatioType.SR_16_TO_9;
117         }
118         else if(Math.abs(1.6-ratio)<0.01){
119             return ScreenRatioType.SR_16_TO_10;
120         }
121         else if(Math.abs(1.666-ratio)<0.01){
122             return ScreenRatioType.SR_5_TO_3;
123         }
124         else if(Math.abs(1.333-ratio)<0.01){
125             return ScreenRatioType.SR_4_TO_3;
126         }
127         else if(Math.abs(1.5-ratio)<0.01){
128             return ScreenRatioType.SR_3_TO_2;
129         }
130         
131         return ScreenRatioType.UNKNOWN;
132     }
