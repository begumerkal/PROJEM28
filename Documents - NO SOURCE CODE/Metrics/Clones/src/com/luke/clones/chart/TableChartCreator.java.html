29  public class TableChartCreator {
30      
31      protected int chartWidth;
32      protected int chartHeight;
33  
34      private Texture part1Texture;
35      private Texture part2Texture;
36      private Texture part3Texture;
37      
38      protected TableChart tableChart;
39      
40      public TableChartCreator() {
41          
42      }
43      
44      public void setChartSize(int chartWidth, int chartHeight) {
45          this.chartWidth = chartWidth;
46          this.chartHeight = chartHeight;
47      }
48      
49      public void setTextures(Texture part1Texture, Texture part2Texture, Texture part3Texture){
50          this.part1Texture = part1Texture;
51          this.part2Texture = part2Texture;
52          this.part3Texture = part3Texture;
53      }
54      
55      public TableChart createWithPercentScale(int part1Percent, int part2Percent, int part3Percent){
56          part1Percent = preventPercentValueFromBeingNegativeOrBiggerThan100(part1Percent);
57          part2Percent = preventPercentValueFromBeingNegativeOrBiggerThan100(part2Percent);
58          part3Percent = preventPercentValueFromBeingNegativeOrBiggerThan100(part3Percent);
59          
60          checkInitValues();
61          
62          int part1Width = calculatePartWidthWithPercentScale(part1Percent);
63          int part2Width = calculatePartWidthWithPercentScale(part2Percent);
64          int part3Width = calculatePartWidthWithPercentScale(part3Percent);
65          
66          part1Width=trimWidthIfTooWide(part1Width, part2Width, part3Width);
67  
68          createNewTableChart();
69          
70          drawParts(part1Width, part2Width, part3Width);
71          
72          return tableChart;
73      }
74      
75      private int preventPercentValueFromBeingNegativeOrBiggerThan100(int percentValue){
76          percentValue = Math.abs(percentValue);
77          if(percentValue>100) percentValue = 100;
78          return percentValue;
79      }
80      
81      protected void checkInitValues(){
82          boolean texturesNotInitialized = part1Texture==null || part2Texture==null || part3Texture==null;
83          boolean chartSizeNotInitialized = chartWidth<=0 || chartHeight<=0;
84          if(texturesNotInitialized || chartSizeNotInitialized){
85              throw new ChartCreatorNotInitializedException();
86          }
87      }
88      
89      protected int calculatePartWidthWithPercentScale(int partPercent){
90          int partWidth = Math.round((float)partPercent/100f*(float)chartWidth);
91          return partWidth;
92      }
93      
94      protected int trimWidthIfTooWide(int part1Width, int part2Width, int part3Width){
95          int widthDifference = chartWidth - (part1Width+part2Width+part3Width);
96          part1Width+=widthDifference;
97          return part1Width;
98      }
99      
100     private void createNewTableChart(){
101         tableChart = new TableChart();
102     }
103     
104     protected void drawParts(int part1Width, int part2Width, int part3Width){
105         final TextureRegion part1TextureRegion = new TextureRegion(part1Texture, 0, 0, part1Width, chartHeight);
106         final TextureRegion part2TextureRegion = new TextureRegion(part2Texture, 0, 0, part2Width, chartHeight);
107         final TextureRegion part3TextureRegion = new TextureRegion(part3Texture, 0, 0, part3Width, chartHeight);
108         
109         tableChart.clear();
110         
111         Image part1Image = new Image(part1TextureRegion);
112         Image part2Image = new Image(part2TextureRegion);
113         Image part3Image = new Image(part3TextureRegion);
114         
115         tableChart.add(part1Image);
116         tableChart.add(part2Image);
117         tableChart.add(part3Image);
118     }
119 
120 }
