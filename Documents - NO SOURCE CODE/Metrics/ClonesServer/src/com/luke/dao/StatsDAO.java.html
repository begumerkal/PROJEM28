49      public GeneralStatsTypeDTO getStatisticTypeByStatsType(StatsType statsType){
50          try{
51              //
52              connection = databaseService.connectToDatabaseIfNotConnectedAndReturnConnection();
53              
54              String selectStatsticStatement = "select id, type from general_stats_type where type=?";
55              PreparedStatement preparedStatement = connection.prepareStatement(selectStatsticStatement);
56              preparedStatement.setString(1, statsType.name());
57              ResultSet resultSet = preparedStatement.executeQuery();
58              while(resultSet.next()){
59                  Long resultId = resultSet.getLong("id");
60                  String resultType = resultSet.getString("type");
61                  
62                  return new GeneralStatsTypeDTO(resultId, statsType);
63              }
64              return null;
65          } catch (SQLException e) {
66              log.error("An error error occured during sql statement execution");
67              log.error("",e);
68              return null;
69          }
70      }

184     private void updateStatisticValue(GeneralStatsDTO statistic){
185         try{
186             connection = databaseService.connectToDatabaseIfNotConnectedAndReturnConnection();
187             
188             String updateStatisticStatement = "update general_stats set value=?, general_stats_type=?, _user=? where id=?";
189             PreparedStatement preparedStatement = connection.prepareStatement(updateStatisticStatement);
190             preparedStatement.setLong(1, statistic.getValue());
191             preparedStatement.setLong(2, statistic.getGeneral_stats_type());
192             preparedStatement.setLong(3, statistic.getUser());
193             preparedStatement.setLong(4, statistic.getId());
194             preparedStatement.executeUpdate();
195         } catch (SQLException e) {
196             log.error("An error error occured during sql statement execution");
197             log.error("",e);
198         }
199     }
