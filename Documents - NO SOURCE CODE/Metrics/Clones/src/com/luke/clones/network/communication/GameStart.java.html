25  import com.luke.clones.model.Map;
26  import com.luke.clones.model.type.PlayerType;
27  
28  public class GameStart {
29      private Map map;
30      private PlayerType playerColor;
31      private Token token;
32      private RoomInfo roomInfo;
33      private PlayerType[] playerTypeOrder;
34      
35      public GameStart(){
36          
37      }
38  
39      public GameStart(Map map, PlayerType playerColor, Token token, RoomInfo roomInfo, PlayerType[] playerTypeOrder) {
40          this.map = map;
41          this.playerColor = playerColor;
42          this.token = token;
43          this.roomInfo = roomInfo;
44          this.playerTypeOrder = playerTypeOrder;
45      }
46  
47      public Map getMap() {
48          return map;
49      }
50  
51      public PlayerType getPlayerColor() {
52          return playerColor;
53      }
54  
55      public Token getToken() {
56          return token;
57      }
58  
59      public RoomInfo getRoomInfo() {
60          return roomInfo;
61      }
62  
63      public PlayerType[] getPlayerTypeOrder() {
64          return playerTypeOrder;
65      }
66      
67  }
