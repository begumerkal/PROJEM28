66      private boolean showRateItButton;
67      
68      private AchievementType(String title, AchievementGenreType achievementGenreType, int value) {
69          this.title = title;
70          this.achievementGenreType = achievementGenreType;
71          this.value = value;
72          this.showRateItButton = true;
73      }
74      
75      private AchievementType(String title, AchievementGenreType achievementGenreType, int value, boolean showRateItButton) {
76          this.title = title;
77          this.achievementGenreType = achievementGenreType;
78          this.value = value;
79          this.showRateItButton = showRateItButton;
80      }

98      public String getDescription(){
99          if(achievementGenreType==AchievementGenreType.PLAY_TIME){
100             if(value==1){
101                 return "Play Clones for "+value+" hour";
102             }
103             else{
104                 return "Play Clones for "+value+" hours";
105             }
106         }
107         else if(achievementGenreType==AchievementGenreType.GAMES_PLAYED){
108             return "Play "+value+" games";
109         }
110         else if(achievementGenreType==AchievementGenreType.GAMES_WON){
111             return "Win "+value+" games";
112         }
113         else if(achievementGenreType==AchievementGenreType.GAMES_WON_IN_ROW){
114             return "Win "+value+" games in a row";
115         }
116         else{
117             return "";
118         }
119     }

121     public static AchievementType[] valuesByGenre(AchievementGenreType achievementGenreType){
122         List<AchievementType> selectedAchievements = new ArrayList<AchievementType>();
123         for(AchievementType achievementType: AchievementType.values()){
124             if(achievementType.getAchievementGenreType()==achievementGenreType){
125                 selectedAchievements.add(achievementType);
126             }
127         }
128         return selectedAchievements.toArray(new AchievementType[0]);
129     }
