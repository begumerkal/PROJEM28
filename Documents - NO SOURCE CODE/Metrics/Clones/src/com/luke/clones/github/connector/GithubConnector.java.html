33  //connector for sending bugs/issues as issue to Github account using Github API

37  
38      }

40      public void sendIssue(String title, String body) throws BadResponseCodeException, IOException {
41              URL url = new URL(
42                      "https://api.github.com/repos/reponame/reposubname/issues");// !!! PUT YOUR OWN REPO URL THERE !!! 
43              HttpURLConnection conn = (HttpURLConnection) url.openConnection();
44              conn.setRequestMethod("POST");
45              conn.setRequestProperty("Accept", "application/json");
46              conn.setRequestProperty(
47                      "Authorization",
48                      "Basic XXXXXXXXXXXXXXXXXXXXXXX"); // !! PUT YOUR OWN API KEY THERE !!!
49              conn.setDoOutput(true);
50  
51              String jsonMessage = "{\"title\": \""+title+"\",\"body\": \""+body+"\"}";
52  
53              BufferedWriter out = new BufferedWriter(new OutputStreamWriter(
54                      conn.getOutputStream()));
55              out.write(jsonMessage);
56              out.close();
57              
58              if (conn.getResponseCode() != 201) {
59                  BufferedReader br = new BufferedReader(new InputStreamReader(
60                          (conn.getInputStream())));
61                  String fullOutput="";
62                  String output;
63                  System.out.println("Output from Server .... \n");
64                  while ((output = br.readLine()) != null) {
65                      fullOutput+=output;
66                  }
67                  throw new BadResponseCodeException("Failed : HTTP error code : "
68                          + conn.getResponseCode() + "Output from Server: "+ fullOutput);
69              }
70              
71              conn.disconnect();
72      }
73  }
