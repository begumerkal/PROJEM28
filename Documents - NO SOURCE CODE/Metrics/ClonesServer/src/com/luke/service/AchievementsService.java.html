40  public class AchievementsService {
41      public static final AchievementsService INSTANCE = new AchievementsService();
42      
43      private Logger log = Logger.getLogger(AchievementsService.class);
44      private UsersDAO usersDAO = UsersDAO.INSTANCE;
45      private StatsGetService statsGetService = StatsGetService.INSTANCE;
46      private AchievementsDAO achievementsDAO = AchievementsDAO.INSTANCE;
47

49

52      public List<AchievementType> updateUserAchievements(String userGoogleId){
53          log.info("Updating achievements for user with google_id "+userGoogleId);
54          List<AchievementType> newAchievements = new ArrayList<AchievementType>();
55          
56          UserDTO userDTO = usersDAO.getUserByGoogleId(userGoogleId);
57          
58          long playTime = getUserPlayTime(userGoogleId);
59          newAchievements.addAll(getNotYetAddedUserAchievementsByAcievementGenreType(userDTO, AchievementGenreType.PLAY_TIME, playTime));
60          long gamesPlayed = getUserPlayedGames(userGoogleId);
61          newAchievements.addAll(getNotYetAddedUserAchievementsByAcievementGenreType(userDTO, AchievementGenreType.GAMES_PLAYED, gamesPlayed));
62          long gamesWon = getUserWonGames(userGoogleId);
63          newAchievements.addAll(getNotYetAddedUserAchievementsByAcievementGenreType(userDTO, AchievementGenreType.GAMES_WON, gamesWon));
64          long gamesWonInRow = getUserBiggestWonInRow(userGoogleId);
65          newAchievements.addAll(getNotYetAddedUserAchievementsByAcievementGenreType(userDTO, AchievementGenreType.GAMES_WON_IN_ROW, gamesWonInRow));
66          
67          for(AchievementType newAchievement: newAchievements){
68              AchievementTypeDTO achievementTypeDTO = achievementsDAO.getAchievementTypeByAchievementType(newAchievement);
69              AchievementDTO achievementToInsert = new AchievementDTO(0L, achievementTypeDTO.getId(), userDTO.getId());
70              achievementsDAO.insertAchievementValue(achievementToInsert);
71          }
72          
73          log.info("Successfully updated achievements for user with google_id "+userGoogleId);
74          return newAchievements;
75      }
76      
77      public List<AchievementType> getUserAchievements(String userGoogleId){
78          log.info("Getting achievements for user with google_id "+userGoogleId);
79          List<AchievementType> userAchievements = new ArrayList<AchievementType>();
80          
81          UserDTO userDTO = usersDAO.getUserByGoogleId(userGoogleId);
82          
83          List<AchievementTypeDTO> userAchievementTypeDTOs = achievementsDAO.getUserAchievementTypes(userDTO.getId());
84          for(AchievementTypeDTO achievementTypeDTO: userAchievementTypeDTOs){
85              userAchievements.add(achievementTypeDTO.getAchievementType());
86          }
87          
88          log.info("Successfully got achievements for user with google_id "+userGoogleId);
89          return userAchievements;
90      }
91      
92      private List<AchievementType> getNotYetAddedUserAchievementsByAcievementGenreType(UserDTO userDTO, AchievementGenreType achievementGenreType, long currentValue){
93          log.info("Obtaining new achievements of type "+achievementGenreType+" for user with google_id "+userDTO.getGoogleId());
94          List<AchievementType> newAchievements = new ArrayList<AchievementType>();
95          
96          for(AchievementType achievementType: AchievementType.valuesByGenre(achievementGenreType)){
97              if(currentValue>=achievementType.getValue()){
98                  AchievementTypeDTO achievementTypeDTO = achievementsDAO.getAchievementTypeByAchievementType(achievementType);
99                  AchievementDTO achievement = achievementsDAO.getAchievementByUserAndAchievementsType(userDTO.getId(), achievementTypeDTO.getId());
100                 if(achievement==null){
101                     newAchievements.add(achievementType);
102                 }
103             }
104         }
105         
106         log.info("Successfully obtained new achievements of type "+achievementGenreType+" for user with google_id "+userDTO.getGoogleId());
107         return newAchievements;
108     }

110     private long getUserPlayTime(String userGoogleId){
111         log.info("Calculating current total play time for user with google_id "+userGoogleId);
112         long playTimeMs=statsGetService.getGeneralStatValue(userGoogleId, StatsType.TOTAL_PLAY_TIME)*1000;
113         long playTimeHours = TimeUnit.MILLISECONDS.toHours(playTimeMs);
114         log.info("Successfully calculated current total play time for user with google_id "+userGoogleId);
115         return playTimeHours;
116     }
117     
118     private long getUserPlayedGames(String userGoogleId){
119         log.info("Calculating current total played games for user with google_id "+userGoogleId);
120         long playedGames = 0;
121         playedGames+=statsGetService.getGeneralStatValue(userGoogleId, StatsType.PLAYED_2P_GAMES);
122         playedGames+=statsGetService.getGeneralStatValue(userGoogleId, StatsType.PLAYED_4P_GAMES);
123         log.info("Successfully calculated current total played games for user with google_id "+userGoogleId);
124         return playedGames;
125     }
126     
127     private long getUserWonGames(String userGoogleId){
128         log.info("Calculating current total won games for user with google_id "+userGoogleId);
129         long wonGames = 0;
130         wonGames+=statsGetService.getGeneralStatValue(userGoogleId, StatsType.WON_2P_GAMES);
131         wonGames+=statsGetService.getGeneralStatValue(userGoogleId, StatsType.WON_4P_GAMES);
132         log.info("Successfully calculated current total won games for user with google_id "+userGoogleId);
133         return wonGames;
134     }
135     
136     private long getUserBiggestWonInRow(String userGoogleId){
137         log.info("Calculating current biggest won in row for user with google_id "+userGoogleId);
138         long biggestWonInRow=statsGetService.getGeneralStatValue(userGoogleId, StatsType.BIGGEST_WON_IN_ROW);
139         log.info("Successfully calculated current biggest won in row for user with google_id "+userGoogleId);
140         return biggestWonInRow;
141     }
