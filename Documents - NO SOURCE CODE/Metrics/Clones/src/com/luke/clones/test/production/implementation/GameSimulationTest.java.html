38  public class GameSimulationTest extends ProductionTest implements
39          ListenerService {
40      private final static int S_TO_MS_MULTIPLER = 1000;
41  
42      private int numberOfGames = 49;
43      private int playersEachRoom = 2; // do not change, 4 not implemented yet
44      private int numberOfPlayers;
45  
46      private int nextRoomCreationIntervalInSeconds = 1;
47  
48      private volatile List<TestRoomInfoWrapper> testRooms;
49  
50      private TimerTask startTask;
51  
52      public GameSimulationTest(AndroidNativeHandler androidNativeHandler,
53              int serverNumberOnServerList) {
54          super(androidNativeHandler, serverNumberOnServerList);
55          numberOfPlayers = numberOfGames * playersEachRoom;
56          super.setNumberOfClients(numberOfPlayers);
57      }
58  
59      @Override
60      protected void test() {
61          initialize();
62          prepareAndRunStartTask();
63      }
64  
65      private void initialize() {
66          testRooms = new ArrayList<TestRoomInfoWrapper>();
67      }
68  
69      private void prepareAndRunStartTask() {
70          startTask = new TimerTask() {
71              @Override
72              public void run() {
73                  startLoop();
74              }
75          };
76  
77          Timer startTimer = new Timer();
78          startTimer.schedule(startTask, 0, nextRoomCreationIntervalInSeconds
79                  * S_TO_MS_MULTIPLER);
80      }
81  
82      private void startLoop() {
83          try {
84              if (testRooms.size() < numberOfGames) {
85                  int j;
86                  if (testRooms.size() == 0)
87                      j = (testRooms.size()) * 2;
88                  else {
89                      j = (testRooms.size() - 1) * 2;
90                  }
91                  final TestRoomInfoWrapper testRoom = new TestRoomInfoWrapper();
92                  final ClonesClient clientAdmin = clonesClientTestWrappers
93                          .get(j).getClonesClient();
94                  final ClonesClient clientB = clonesClientTestWrappers
95                          .get(j + 1).getClonesClient();
96                  testRoom.createRoom(clientAdmin);
97                  Thread.sleep(100L);
98                  testRoom.joinRoom(clientB);
99                  Thread.sleep(100L);
100                 testRoom.startGame(clientAdmin);
101                 testRooms.add(testRoom);
102 
103                 new Timer().schedule(new TimerTask() {
104 
105                     @Override
106                     public void run() {
107                         // System.out.println("A");
108                         testRoom.skipTurn(clientAdmin);
109                     }
110                 }, 3000, 6000);
111                 new Timer().schedule(new TimerTask() {
112 
113                     @Override
114                     public void run() {
115                         // System.out.println("B");
116                         testRoom.skipTurn(clientB);
117                     }
118                 }, 6000, 6000);
119             } else {
120                 startTask.cancel();
121             }
122         } catch (InterruptedException e) {
123             // TODO Auto-generated catch block
124             e.printStackTrace();
125         }
126     }
127 
128     @Override
129     protected void stopTest() {
130         try {
131             startTask.cancel();
132             Thread.sleep(300L);
133             for (ClonesClientTestWrapper clonesClientTestWrapper : clonesClientTestWrappers) {
134                 clonesClientTestWrapper.getClonesClient().stop();
135                 System.out.println("One of clients stopped");
136                 Thread.sleep(100L);
137             }
138         } catch (InterruptedException e) {
139             // TODO Auto-generated catch block
140             e.printStackTrace();
141         }
142     }

148     }
