38  public class AuthorizationUtil {
39      
40      private static String googleUserDataString="No_data";
41      
42      public static TokenValidation isTokenValid(String token) throws IOException{
43          URL url = new URL("https://www.googleapis.com/oauth2/v1/userinfo?access_token="+token);
44          HttpURLConnection con = (HttpURLConnection) url.openConnection();
45          int sc = con.getResponseCode();
46          if (sc == 200) {
47              InputStream is = con.getInputStream();
48              googleUserDataString = readResponse(is);
49              is.close();
50              System.out.println("token ok");
51              System.out.println("User data:"+googleUserDataString);
52              GoogleUserData googleUserDataObject = null;
53              try {
54                  googleUserDataObject = decodeGoogleUserDataJson(googleUserDataString);
55              } catch (ParseException e) {
56                  e.printStackTrace();
57              }
58              return new TokenValidation(true, googleUserDataObject);
59          } else if (sc == 401) {
60              //TODO: to robiæ na frontendzie jak siê nie powiedzie autentykacja
61  //          GoogleAuthUtil.clearToken(activity, token);
62              System.out.println("token invalid");
63              return new TokenValidation(false, null);
64          } else {
65              System.out.println("token invalid (some other error occured)");
66              return new TokenValidation(false, null);
67          }
68      }
69      
70      private static String readResponse(InputStream is) throws IOException {
71          ByteArrayOutputStream bos = new ByteArrayOutputStream();
72          byte[] data = new byte[2048];
73          int len = 0;
74          while ((len = is.read(data, 0, data.length)) >= 0) {
75              bos.write(data, 0, len);
76          }
77          return new String(bos.toByteArray(), "UTF-8");
78      }
79      
80      private static GoogleUserData decodeGoogleUserDataJson(String googleUserDataJson) throws ParseException{
81          JSONParser parser=new JSONParser();
82          JSONObject jsonObject=null;
83          jsonObject=(JSONObject) parser.parse(googleUserDataJson);
84          GoogleUserData googleUserData = new GoogleUserData();
85          googleUserData.setId((String) jsonObject.get("id"));
86          googleUserData.setName((String) jsonObject.get("name"));
87          googleUserData.setGivenName((String) jsonObject.get("given_name"));
88          googleUserData.setFamilyName((String) jsonObject.get("family_name"));
89          googleUserData.setLink((String) jsonObject.get("link"));
90          googleUserData.setPicture((String) jsonObject.get("picture"));
91          googleUserData.setGender((String) jsonObject.get("gender"));
92          googleUserData.setLocale((String) jsonObject.get("locale"));
93          return googleUserData;
94      }
