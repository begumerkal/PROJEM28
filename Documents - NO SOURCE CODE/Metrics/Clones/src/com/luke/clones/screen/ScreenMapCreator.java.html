212     private void createBoard(int fieldsInWidth, int fieldsInHeight) {
213         
214         int screenWidth = Gdx.graphics.getWidth() - Config.widthOffset;
215         int screenHeight = Gdx.graphics.getHeight() - Config.gameHeightOffset - Config.timeBarHeight;
216         
217         fieldListener = new FieldListener();
218         
219         int xFieldSize = screenWidth/fieldsInHeight;
220         int yFieldSize = screenHeight/fieldsInWidth;
221         
222         if(xFieldSize < yFieldSize) fieldSize = xFieldSize;
223         else fieldSize = yFieldSize;
224         unusedSpace = (int)mainStage.getWidth() - (fieldSize * fieldsInHeight);
225     }
226     
227     private class FieldListener extends ClickListener{
228         boolean click = false;
229 
230         @Override
231         public void enter(InputEvent event, float x, float y, int pointer,
232                 Actor fromActor) {
233             if(click==true){
234                 FieldActor2 field = (FieldActor2) event.getListenerActor();
235                 if(field.getFieldType()!=DrawFieldButtonTypeChanger.getCurrent().getDrawFieldType()){
236                     field.setFieldType(DrawFieldButtonTypeChanger.getCurrent().getDrawFieldType());
237                     if(DrawFieldButtonTypeChanger.getCurrent().getDrawFieldType()==FieldType.TAKEN){
238                         field.setColor(DrawFieldButtonTypeChanger.getCurrent().getDrawFieldColor());
239                     }
240                     else{
241                         field.setColor(ERASE_FIELD_COLOR);
242                     }
243                 }
244                 else{
245                     field.setFieldType(ERASE_FIELD_TYPE);
246                     field.setColor(ERASE_FIELD_COLOR);
247                 }
248                 
249             }
250         }
251 
252         @Override
253         public boolean touchDown(InputEvent event, float x, float y,
254                 int pointer, int button) {
255             SoundManager.INSTANCE.play(SoundType.CLICK);
256             click=true;
257             return true;
258         }
259 
260         @Override
261         public void touchUp(InputEvent event, float x, float y, int pointer,
262                 int button) {
263             click=false;
264         }
265         
266         
267 
268     }

270     private void buildStage(int fieldsInWidth, int fieldsInHeight){
271         if(fieldMap!=null){
272             for(FieldActor2 fieldActor: fieldMap.values()){
273                 fieldActor.remove();
274             }
275         }
276         
277         fieldMap = new HashMap<Position, FieldActor2>();
278         
279         Texture emptyTexture = new Texture(Gdx.files.internal("data/models/blank.png"));
280         Texture solidTexture = new Texture(Gdx.files.internal("data/models/solid.png"));
281         Texture ballTexture = new Texture(Gdx.files.internal("data/models/ball.png"));
282         
283         for(int i=0; i<fieldsInWidth;i++){
284             for(int j=0; j<fieldsInHeight;j++){
285                     FieldActor2 emptyField = new FieldActor2(emptyTexture, solidTexture, ballTexture, emptyTexture, null, new Position(i,j),FieldType.EMPTY);
286                     emptyField.setPosition((j*fieldSize)+unusedSpace/2, i*fieldSize);
287                     emptyField.setSize(fieldSize, fieldSize);
288                     emptyField.addListener(fieldListener);
289                     mainStage.addActor(emptyField);
290                     fieldMap.put(emptyField.getPosition(), emptyField);
291             }
292         }
293     }

320     private PlayerType fieldColorToPlayerType(Color color){
321             if(color.equals(Color.BLUE)) {
322                 return PlayerType.BLUE;
323             }
324             else if(color.equals(Color.GREEN)){
325                 return PlayerType.GREEN;
326             }
327             else if(color.equals(Color.RED)){
328                 return PlayerType.RED;
329             }
330             else if(color.equals(Color.ORANGE)){
331                 return PlayerType.ORANGE;
332             }
333             else{
334                 return null;
335             }
336     }
