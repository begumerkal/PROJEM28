35  public class MusicManager {
36      private static final Integer START_TRACK = 0;
37      private static final float VOLUME = 0.07f;
38  //  private static final long DELAY_MS = 500;
39      
40      private static final long VOLUME_CHANGE_ACCURACY_MS = 10; //The smaller the bigger performance load
41      
42      private static final long GRADUALLY_INCREASE_VOLUME_TIME = 1500L;
43      private static final float GRADUALLY_INCREASE_VOLUME_TIME_SINGLE_ITERATION_VOLUME_CHANGE = (VOLUME/GRADUALLY_INCREASE_VOLUME_TIME) * VOLUME_CHANGE_ACCURACY_MS;
44      
45      private static final long GRADUALLY_MUTE_TIME = 1000L;
46      private static final float GRADUALLY_MUTE_TIME_SINGLE_ITERATION_VOLUME_CHANGE = (VOLUME/GRADUALLY_MUTE_TIME) * VOLUME_CHANGE_ACCURACY_MS;
47  
48      public static final MusicManager INSTANCE = new MusicManager();
49  
50      private LinkedHashMap<Integer, Music> musicMap;
51  
52      private Timer timer;
53      
54      private Timer graduallyDecreaseVolumeTimer;
55  
56      private Integer currentTrack = START_TRACK;
57      
58      private boolean gradualVolumeIncreaseBeforePlay = false;
59  
60      protected MusicManager() {
61          timer = new Timer();
62          graduallyDecreaseVolumeTimer = new Timer();
63  
64          musicMap = new LinkedHashMap<Integer, Music>();
65          musicMap.put(0, Gdx.audio.newMusic(Gdx.files
66                  .internal("data/music/epic_orchestra.ogg")));
67  
68          for (Music music : musicMap.values()) {
69              music.setOnCompletionListener(new OnCompletionListener() {
70                  @Override
71                  public void onCompletion(Music music) {
72                      playNextTrack();
73                  }
74              });
75          }
76      }
77  
78      private void playNextTrack() {
79          currentTrack++;
80          if (currentTrack >= musicMap.size()) {
81              currentTrack = 0;
82          }
83          play();
84      }
85  
86      public void play() {
87          if (SettingsManager.getSettingValue(SettingType.MUSIC_ON) == true) {
88              try {
89                  timer.cancel();
90                  graduallyDecreaseVolumeTimer.cancel();
91              } catch (IllegalStateException e) {
92  
93              }
94              if(gradualVolumeIncreaseBeforePlay==false){
95                  playImmediately();
96              }
97              else{
98                  graduallyIncreaseVolumeAndPlay();
99              }
100         }
101     }
102 
103      private void playImmediately(){
104          musicMap.get(currentTrack).setVolume(VOLUME);
105          musicMap.get(currentTrack).play();
106      }
107     
108     private void graduallyIncreaseVolumeAndPlay(){
109         gradualVolumeIncreaseBeforePlay = false;
110         musicMap.get(currentTrack).setVolume(0);
111         musicMap.get(currentTrack).play();
112         TimerTask timerTask = new TimerTask() {
113             @Override
114             public void run() {
115                 if(musicMap.get(currentTrack).getVolume()<VOLUME){
116                     float currentVolume = musicMap.get(currentTrack).getVolume();
117                     musicMap.get(currentTrack).setVolume(currentVolume+GRADUALLY_INCREASE_VOLUME_TIME_SINGLE_ITERATION_VOLUME_CHANGE);
118                 }
119                 else{
120                     timer.cancel();
121                 }
122             }
123         };
124 
125         timer = new Timer();
126         timer.schedule(timerTask, 0, VOLUME_CHANGE_ACCURACY_MS);
127     }
128 
129     public void pauseImmediately() {
130         try {
131             timer.cancel();
132             graduallyDecreaseVolumeTimer.cancel();
133         } catch (IllegalStateException e) {
134 
135         }
136         musicMap.get(currentTrack).pause();
137     }
138     
139     private void graduallyDecreaseVolumeAndPause(){
140         try {
141             timer.cancel();
142             graduallyDecreaseVolumeTimer.cancel();
143         } catch (IllegalStateException e) {
144 
145         }
146         TimerTask timerTask = new TimerTask() {
147             @Override
148             public void run() {
149                 if(musicMap.get(currentTrack).getVolume()>0){
150                     float currentVolume = musicMap.get(currentTrack).getVolume();
151                     musicMap.get(currentTrack).setVolume(currentVolume-GRADUALLY_MUTE_TIME_SINGLE_ITERATION_VOLUME_CHANGE);
152                 }
153                 else{
154                     musicMap.get(currentTrack).pause();
155                     graduallyDecreaseVolumeTimer.cancel();
156                 }
157             }
158         };
159 
160         graduallyDecreaseVolumeTimer = new Timer();
161         graduallyDecreaseVolumeTimer.schedule(timerTask, 0, VOLUME_CHANGE_ACCURACY_MS);
162     }
163 
164     public void pauseAndScheduleGradualVolumeIncreaseBeforePlay() {
165         try {
166             timer.cancel();
167         } catch (IllegalStateException e) {
168 
169         }
170         graduallyDecreaseVolumeAndPause();
171         gradualVolumeIncreaseBeforePlay=true;
172     }
173     
174     public void dispose() {
175         try {
176             timer.cancel();
177         } catch (IllegalStateException e) {
178 
179         }
180         for (Music music : musicMap.values()) {
181             music.dispose();
182         }
183         musicMap.clear();
184     }
185 }
