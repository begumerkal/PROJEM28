89      @Override
90      public void received(Connection connection, Object obj) {
91          if(obj.getClass()!=KeepAlive.class){
92              log.debug("Message received. Msg type: "+obj.getClass().toString()+". Connection id: "+connection.getID());
93          }
94          if(obj instanceof GetServerBackendInfo){
95              GetServerBackendInfo message = (GetServerBackendInfo) obj;
96              connection.sendTCP(new ServerBackendInfo(playerList.size(), Config.maxNumberOfPlayers));
97          }
98          else if(obj instanceof MoveLogin){
99              try {
100                 MoveLogin login = (MoveLogin) obj;
101                 String token = login.getToken();
102                 
103                 TokenValidation tokenValidation = new TokenValidation(false, null);
104                 //Should be commented on production level for security reasons
105                 if(Config.testModeOn && token.equals("pcUKwZZuUpmMajlTWlgLhVwpskzybSaqlmBZcQKpYOIMNGzUka")){
106                     GoogleUserData googleUser = new GoogleUserData();
107                     googleUser.setId("1");
108                     tokenValidation = new TokenValidation(true, googleUser);
109                 }
110                 else{
111                     tokenValidation = AuthorizationUtil.isTokenValid(token);
112                 }
113                 boolean tokenValid = tokenValidation.isTokenValid();
114                 
115                 String playerName = login.getPlayerName();
116                 
117                 if(playerList.size()>=Config.maxNumberOfPlayers){
118                     connection.sendTCP(new UpdateLogin(ActionTypeUpdateLogin.SERVER_FULL));
119                     return;
120                 }
121                 
122                 if(playerName.length()==0){
123 //                  connection.sendTCP(new UpdateLogin(ActionTypeUpdateLogin.FORBIDDEN_NAME));
124                     connection.sendTCP(new UpdateLogin(ActionTypeUpdateLogin.UNEXPECTED_ERROR));
125                     return;
126                 }
127                 else if(playerList.contains(playerName)==true){
128 //                  connection.sendTCP(new UpdateLogin(ActionTypeUpdateLogin.NAME_ALREADY_TAKEN));
129                     connection.sendTCP(new UpdateLogin(ActionTypeUpdateLogin.UNEXPECTED_ERROR));
130                     return;
131                 }
132                 
133                 if(tokenValid){
134                     playerList.add(new Player(playerName, tokenValidation.getUserData(), connection));
135                     connection.sendTCP(new UpdateLogin(ActionTypeUpdateLogin.SUCCESS));
136                     log.info("Player \""+login.getPlayerName()+ "\" successfully logged in");
137                     usersService.addUserIfNotPresent(tokenValidation.getUserData().getId(), login.getPlayerName());
138                 }
139                 else{
140                     //for test only
141                     connection.sendTCP(new UpdateLogin(ActionTypeUpdateLogin.INVALID_AUTH_TOKEN));
142                 }
143             } catch (IOException e) {
144                 connection.sendTCP(new UpdateLogin(ActionTypeUpdateLogin.UNEXPECTED_ERROR));
145                 log.error("", e);
146             }
147             
148         }
149         else if(obj instanceof MoveRoom){
150             MoveRoom moveRoom = (MoveRoom) obj;
151             ActionTypeMoveRoom action = moveRoom.getActionType();
152             if(action == ActionTypeMoveRoom.GET_LIST){
153                 //sending room list to client
154                 ActionTypeUpdateRoom actionType = ActionTypeUpdateRoom.ROOM_LIST;
155                 RoomInfo[] roomInfoList = roomList.getRoomInfoList();
156                 RoomInfo roomInterior = null;//we need to send only list this time
157                 UpdateRoom updateRoom = new UpdateRoom(actionType, roomInfoList, roomInterior);
158                 connection.sendTCP(updateRoom);
159             }
160             if(action == ActionTypeMoveRoom.CREATE){
161                 RoomCreateInfo roomCreateInfo = moveRoom.getRoomCreateInfo();
162                 String tempRoomName = moveRoom.getRoomName();
163                 if(tempRoomName.length()==0){
164                     connection.sendTCP(new UpdateRoom(ActionTypeUpdateRoom.FORBIDDEN_NAME, null, null));
165                 }
166                 else if(roomList.contains(tempRoomName)==true){
167                     connection.sendTCP(new UpdateRoom(ActionTypeUpdateRoom.NAME_ALREADY_TAKEN, null, null));
168                 }
169                 else{
170                     Room tempRoom = new Room(tempRoomName, playerList.get(connection));
171                     tempRoom.setRoomCreateInfo(roomCreateInfo);
172                     roomList.add(tempRoom);
173                     connection.sendTCP(new UpdateRoom(ActionTypeUpdateRoom.CREATE_SUCCESS, null, tempRoom.getRoomInfo()));
174                     //sending info to all without room
175                     playerList.sendToAllPlayersWithoutRoom((new UpdateRoom(ActionTypeUpdateRoom.ROOM_LIST, roomList.getRoomInfoList(), null)), roomList);
176                     log.info("Room \""+tempRoomName+"\" created");
177                 }
178             }
179             
180             if(action == ActionTypeMoveRoom.MODIFY_REQUEST){
181                 Room tempRoom = roomList.get(connection);
182                 if(tempRoom!=null){
183                     Player roomAdmin = tempRoom.getAdmin();
184                     if(roomAdmin.getName().equals(playerList.get(connection).getName())){
185                         connection.sendTCP(new UpdateRoom(ActionTypeUpdateRoom.MODIFY_REQUEST_RESPONSE, null,
186                                 tempRoom.getRoomInfo()));
187                     }
188                 }
189             }
190             if(action == ActionTypeMoveRoom.MODIFY){
191                 RoomModifyInfo roomModifyInfo = moveRoom.getRoomUpdateInfo();
192                 Room tempRoom = roomList.get(roomModifyInfo.getRoomName());
193                 if(tempRoom!=null){
194                     Player roomAdmin = tempRoom.getAdmin();
195                     if(roomAdmin.getName().equals(playerList.get(connection).getName())){
196                         tempRoom.setRoomCreateInfo(roomModifyInfo);
197                         connection.sendTCP(new UpdateRoom(ActionTypeUpdateRoom.ROOM_INTERIOR, null, tempRoom.getRoomInfo()));
198                         tempRoom.sendToAllPlayersExceptSender(new UpdateRoom(ActionTypeUpdateRoom.ROOM_INTERIOR, null, tempRoom.getRoomInfo()), playerList.get(connection));
199                         playerList.sendToAllPlayersWithoutRoom((new UpdateRoom(ActionTypeUpdateRoom.ROOM_LIST, roomList.getRoomInfoList(), null)), roomList);
200                     }
201                 }
202             }
203             if(action == ActionTypeMoveRoom.ENTER){
204                 String tempRoomName = moveRoom.getRoomName();
205                 Room tempRoom = roomList.get(tempRoomName);
206                 if(tempRoom==null){
207                     connection.sendTCP(new UpdateRoom(ActionTypeUpdateRoom.NOT_EXIST, null, null));
208                 }
209                 else if(tempRoom.getNumberOfPlayers() == tempRoom.getPlayerLimit()){
210                     connection.sendTCP(new UpdateRoom(ActionTypeUpdateRoom.IS_FULL, null, null));
211                 }
212                 else{
213                     Player tempPlayer = playerList.get(connection);
214                     tempRoom.addPlayer(tempPlayer);
215                     connection.sendTCP(new UpdateRoom(ActionTypeUpdateRoom.ENTER_SUCCESS, null, tempRoom.getRoomInfo()));
216                     //sending info to all other players in this room
217                     tempRoom.sendToAllPlayersExceptSender(new UpdateRoom(ActionTypeUpdateRoom.ROOM_INTERIOR, null, tempRoom.getRoomInfo()), playerList.get(connection));
218                     //sending info to all without room
219                     playerList.sendToAllPlayersWithoutRoom((new UpdateRoom(ActionTypeUpdateRoom.ROOM_LIST, roomList.getRoomInfoList(), null)), roomList);
220                 }
221             }
222             if(action == ActionTypeMoveRoom.START_GAME){
223                 Player tempPlayer = playerList.get(connection);
224                 Room tempRoom = roomList.get(tempPlayer);
225                 if(tempRoom!=null && tempPlayer.getName().equals(tempRoom.getAdmin().getName()) && tempRoom.getNumberOfPlayers()==tempRoom.getPlayerLimit()){
226                     try {
227                         tempRoom.startNewGame();
228                     } catch (MapCreatorArgumentException e) {
229                         log.error("", e);
230                     }
231                 }
232                 else{
233                     log.info("Game start rejected");
234                 }
235             }
236             if(action == ActionTypeMoveRoom.RESTART_GAME){
237                 Player tempPlayer = playerList.get(connection);
238                 Room tempRoom = roomList.get(tempPlayer);
239                 if(tempRoom!=null && tempPlayer.getName().equals(tempRoom.getAdmin().getName()) && tempRoom.getNumberOfPlayers()==tempRoom.getPlayerLimit()){
240                     try {
241                         tempRoom.restartGame();
242                     } catch (MapCreatorArgumentException e) {
243                         log.error("", e);
244                     }
245                 }
246                 else{
247                     log.info("Game start rejected");
248                 }
249             }
250             if(action == ActionTypeMoveRoom.GET_INTERIOR){
251                 Player tempPlayer = playerList.get(connection);
252                 Room tempRoom = roomList.get(tempPlayer);
253                 if(tempRoom!=null) connection.sendTCP(new UpdateRoom(ActionTypeUpdateRoom.ROOM_INTERIOR, null, tempRoom.getRoomInfo()));
254             }
255             if(action == ActionTypeMoveRoom.LEAVE){
256                 Player tempPlayer = playerList.get(connection);
257                 Room tempRoom = roomList.get(tempPlayer);
258                 if(tempRoom!=null){
259                     if(tempPlayer.getName().equals(tempRoom.getAdmin().getName())){
260                         adminLeaveOrDisconnectAction(tempPlayer, tempRoom, LeaveOrDisconnectActionType.LEAVE);
261                     }
262                     else{
263                         nonAdminPlayerLeaveOrDisconnectAction(tempPlayer, tempRoom, LeaveOrDisconnectActionType.LEAVE);
264                     }
265                     tempRoom.setPause(true);
266                 }
267             }
268             if(action == ActionTypeMoveRoom.STOP_GAME){
269                 Player tempPlayer = playerList.get(connection);
270                 Room tempRoom = roomList.get(tempPlayer);
271                 if(tempRoom!=null){
272                     if(tempPlayer.getName().equals(tempRoom.getAdmin().getName())){
273                         Player[] tempRoomPlayers = tempRoom.getAllPlayers();
274                         for(Player tempRoomPlayer: tempRoomPlayers){
275                             if(tempRoomPlayer.getName().equals(tempRoom.getAdmin().getName())){
276                                 tempRoomPlayer.getConnection().sendTCP(new UpdateRoom(ActionTypeUpdateRoom.BACK_TO_ROOM_INTERIOR, null, null));
277                             }
278                             else{
279                                 UpdateRoom updateRoom = new UpdateRoom(ActionTypeUpdateRoom.BACK_TO_ROOM_INTERIOR, null, null);
280                                 updateRoom.setMessageToShow(MessageToShowType.ADMIN_STOPPED);
281                                 tempRoomPlayer.getConnection().sendTCP(updateRoom);
282                             }
283 //                          tempRoomPlayer.getConnection().sendTCP(new UpdateRoom(ActionTypeUpdateRoom.BACK_TO_ROOM_INTERIOR, null, null));
284 //                          tempRoomPlayer.getConnection().sendTCP(new MessageShowRequest(MessageToShowType.ADMIN_STOPPED));
285                         }
286                         tempRoom.setPause(true);
287                     }
288                 }
289             }
290         }
291         else if(obj instanceof MoveBoard){
292             MoveBoard playerMove = (MoveBoard) obj;
293             
294             Room room = roomList.get(connection);
295             PlayerType senderPlayerType = room.getPlayerType(playerList.get(connection));
296             log.info("MoveBoard received, ActionType:"+playerMove.getActionType()+", playerPosition"+playerMove.getPlayerPosition()+", move destination:"+playerMove.getPosition());
297             log.info("Additional Info, room: "+room+", PlayerType: "+senderPlayerType);
298             if(room.isPause()==false) room.playerMove(playerMove, senderPlayerType);
299         }
300         else if(obj instanceof StatsRequest){
301             StatsRequest statsRequest = (StatsRequest) obj;
302             ActionTypeStats actionTypeStats = statsRequest.getActionTypeStats();
303             Player tempPlayer = playerList.get(connection);
304             String tempPlayerGoogleId = tempPlayer.getGoogleUserData().getId();
305             
306             if(actionTypeStats==ActionTypeStats.GET_GENERAL_STATS){
307                 Map<StatsType, Long> generalStatsMap = statsGetService.getAllGeneralStatsValuesMap(tempPlayerGoogleId);
308                 long level = statsGetService.getUserLevel(tempPlayerGoogleId);
309                 long percentToNextLevel = statsGetService.getUserPercentToNextLevel(tempPlayerGoogleId);
310                 StatsResponse statsResponse = new StatsResponse();
311                 statsResponse.setActionTypeStats(ActionTypeStats.GET_GENERAL_STATS);
312                 statsResponse.setStatsMap(generalStatsMap);
313                 statsResponse.setLevel(level);
314                 statsResponse.setPercentToNextLevel(percentToNextLevel);
315                 tempPlayer.getConnection().sendTCP(statsResponse);
316             }
317         }
318         else if(obj instanceof AchievementsRequest){
319             AchievementsRequest achievementRequest = (AchievementsRequest) obj;
320             
321             Player tempPlayer = playerList.get(connection);
322             String tempPlayerGoogleId = tempPlayer.getGoogleUserData().getId();
323             
324             AchievementsResponse achievementsResponse = new AchievementsResponse();
325             achievementsResponse.setCurrentAchievements(achievementsService.getUserAchievements(tempPlayerGoogleId));
326             tempPlayer.getConnection().sendTCP(achievementsResponse);
327             
328         }
329     }

331     @Override
332     public void disconnected(Connection connection) {
333         Player tempPlayer = playerList.get(connection);
334         Room tempRoom = roomList.get(tempPlayer);
335         System.out.println("disconnect");
336         if(tempRoom!=null){
337             System.out.println("disconnect and tempRoom!=null");
338             if(tempPlayer.getName().equals(tempRoom.getAdmin().getName())){
339                 adminLeaveOrDisconnectAction(tempPlayer, tempRoom, LeaveOrDisconnectActionType.DISCONNECT);
340             }
341             else{
342                 nonAdminPlayerLeaveOrDisconnectAction(tempPlayer, tempRoom, LeaveOrDisconnectActionType.DISCONNECT);
343             }
344             tempRoom.setPause(true);
345         }
346         
347         playerList.remove(connection);
348 
349         log.info("player disconnected, connection id: "+connection.getID());
350         log.info("Player list: "+buildPlayerList(playerList.getAllPlayers()));
351     }

353     private void adminLeaveOrDisconnectAction(Player admin, Room room, LeaveOrDisconnectActionType leaveOrDisconnectActionType){
354         room.removePlayer(admin);
355         UpdateRoom updateRoom = new UpdateRoom(ActionTypeUpdateRoom.BACK_TO_ROOM_LIST, null, null);
356         updateRoom = updatePlayerAchievements(updateRoom, admin, leaveOrDisconnectActionType);
357         admin.getConnection().sendTCP(updateRoom);
358         Player[] tempRoomPlayers = room.getAllPlayers();
359         if(tempRoomPlayers.length>0){
360             room.markCurrentPlayerAsAdmin(room.getAllPlayers()[0]);
361             for(Player tempRoomPlayer: tempRoomPlayers){
362                 sendBackToRoomInteriorWithMessage(tempRoomPlayer);
363             }
364         }
365         else{
366             roomList.remove(room);
367         }
368         playerList.sendToAllPlayersWithoutRoom(new UpdateRoom(ActionTypeUpdateRoom.ROOM_LIST, roomList.getRoomInfoList(), null), roomList);
369     }
370     
371     private void nonAdminPlayerLeaveOrDisconnectAction(Player player, Room room, LeaveOrDisconnectActionType leaveOrDisconnectActionType){
372         room.removePlayer(player);
373         UpdateRoom updateRoom = new UpdateRoom(ActionTypeUpdateRoom.BACK_TO_ROOM_LIST, null, null);
374         updateRoom = updatePlayerAchievements(updateRoom, player, leaveOrDisconnectActionType);
375         player.getConnection().sendTCP(updateRoom);
376         Player[] tempRoomPlayers = room.getAllPlayers();
377         for(Player tempRoomPlayer: tempRoomPlayers){
378             sendBackToRoomInteriorWithMessage(tempRoomPlayer);
379         }
380         playerList.sendToAllPlayersWithoutRoom(new UpdateRoom(ActionTypeUpdateRoom.ROOM_LIST, roomList.getRoomInfoList(), null), roomList);
381     }
382     
383     private UpdateRoom updatePlayerAchievements(UpdateRoom updateRoom, Player player, LeaveOrDisconnectActionType leaveOrDisconnectActionType){
384         if(leaveOrDisconnectActionType==LeaveOrDisconnectActionType.LEAVE){
385             List<AchievementType> newAchievements = achievementsService.updateUserAchievements(player.getGoogleUserData().getId());
386             updateRoom.setNewAchievements(newAchievements.toArray(new AchievementType[0]));
387         }
388         return updateRoom;
389     }

391     private void sendBackToRoomInteriorWithMessage(Player tempRoomPlayer) {
392         UpdateRoom updateRoom = new UpdateRoom(ActionTypeUpdateRoom.BACK_TO_ROOM_INTERIOR, null, null);
393         updateRoom.setMessageToShow(MessageToShowType.ONE_OF_PLAYERS_LEFT_ROOM);
394         tempRoomPlayer.getConnection().sendTCP(updateRoom);
395     }
