29  public class MapModelUtils {
30      public static MapModel[] selectMapsWithSpecifiedNumberOfPlayers(int numberOfPlayers){
31          ArrayList<MapModel> selectedMaps = new ArrayList<MapModel>();
32          for(AvailableMapModelsType availableMapModelsType: AvailableMapModelsType.values()){
33              if(availableMapModelsType.getMap().getNumberOfPlayers()==numberOfPlayers){
34                  selectedMaps.add(availableMapModelsType.getMap());
35              }
36          }
37          return selectedMaps.toArray(new MapModel[0]);
38      }
39      
40      public static String[] resolveMapsToNames(MapModel[] mapModels){
41          List<String> nameList = new ArrayList<String>();
42          for(MapModel mapModel: mapModels){
43              nameList.add(mapModel.getMapName());
44          }
45          return nameList.toArray(new String[0]);
46      }
47      
48      public static MapModel resolveNameToMap(String mapName){
49          for(AvailableMapModelsType availableMapModelType : AvailableMapModelsType.values()){
50              MapModel mapModel = availableMapModelType.getMap();
51              if(mapModel.getMapName().equals(mapName)){
52                  return mapModel;
53              }
54          }
55          throw new IllegalArgumentException("Map with name \""+mapName+"\" doesn't exist");
56      }
57  }
