338     private LogicBoard createLogicBoard(int boardWidth, int boardHeight, Position[] startSolids, PlayerPosition[] startPlayers){
339         board = new LogicBoard(boardWidth,boardHeight);
340         
341         if(startSolids != null){
342             for(int i=0; i<startSolids.length; i++){
343                 board.setFieldTypeSolid(startSolids[i].getX(), startSolids[i].getY());
344             }
345         }
346

348             board.setFieldPlayer(startPlayers[i].getX(), startPlayers[i].getY(), startPlayers[i].getPlayerType());

507     private void updateStage(){
508         
509         flashCurrentPlayerBalls();
510         
511         //skips player turn when he is eliminated (when he has 0 balls) and when he has no opportunity to move anywhere
512         if(board.getBallCount(turn.getCurrentPlayer()) == 0
513                 || board.isPlayerMoveable(turn.getCurrentPlayer())==false){
514             
515             //game end condition
516             boolean endGame = true;
517             PlayerType[] availablePlayers = turn.getAvailablePlayers();
518             for(PlayerType player:availablePlayers){
519                 if(player!=turn.getCurrentPlayer()){
520                     if(board.isPlayerMoveable(player)==true) endGame=false;
521                 }
522             }
523             
524             //what program should do at the end
525             if(endGame){
526                 //game Ends Man
527                 pause = true;
528                 
529                 stopMeasureAndSignalizePlayerMoveTime();
530                 layerStage.clear();
531                 
532                 PlayerType[] playerList = turn.getAvailablePlayers();
533                 int[] scoreList = new int[playerList.length];
534                 PlayerScore[] playerScore = new PlayerScore[playerList.length];
535                 for(int i=0; i<scoreList.length;i++){
536                     scoreList[i] = board.getBallCount(playerList[i]);
537                     playerScore[i] = new PlayerScore(playerList[i], scoreList[i]);
538                 }
539                 
540                 Arrays.sort(playerScore, new Comparator<PlayerScore>() {
541                     @Override
542                     public int compare(PlayerScore o1, PlayerScore o2) {
543                         if(o1.getPlayerScore()<o2.getPlayerScore()) return 1;
544                         else if(o1.getPlayerScore()>o2.getPlayerScore()) return -1;
545                         else return 0;
546                     }
547                 });
548                 
549                 boolean firstRound;
550                 if(turnStats==null){
551                     firstRound = true;
552                     try {
553                         turnStats = new TurnStats(playerScore);
554                     } catch (TurnStatsArgumentException e) {
555                         e.printStackTrace();
556                     }
557                 }
558                 else{
559                     firstRound = false;
560                     try {
561                         turnStats.saveTurnStats(playerScore);
562                     } catch (TurnStatsArgumentException e) {
563                         e.printStackTrace();
564                     }
565                 }
566                 
567                 Skin tSkin = new Skin(Gdx.files.internal("data/skins/skin_small.json"));
568                 final Table endTable = new Table(tSkin);
569 //              endTable.setBackground(tSkin.getDrawable("default-round-large"));
570                 endTable.setBackground(tSkin.getDrawable("message-round-large-bright"));

829     private void startCheckingPlayerMoveTime(){
830         if(pause==false){
831             checkingPlayerMoveTimeTask = new TimerTask() {
832                 @Override
833                 public void run() {
834                     float playerMoveTimeNumber = playerMoveTime.getMeasuredTimeInMs()/1000f;
835                     if(playerMoveTimeNumber>playerMoveTimeLimitType.getPlayerMoveTimeLimit().getLimitInSeconds()){
836                         skipTurn();
837                         updateStage();
838                     }
839                 }
840             };
841 
842             checkingPlayerMoveTimeTimer = new Timer(false);
843             checkingPlayerMoveTimeTimer.schedule(checkingPlayerMoveTimeTask,
844                     playerMoveTimeLimitType.getPlayerMoveTimeLimit().getLimitInSeconds()*SECONDS_TO_MILLISECONDS_MULTIPLER+ENSURE_MS);
845         }
846     }

874     private class FieldListener extends ClickListener{
875 
876         @Override
877         public void clicked(InputEvent event, float x, float y) {
878             super.clicked(event, x, y);
879             FieldActor2 field = (FieldActor2) event.getListenerActor();
880             Position pos = field.getPosition();
881             PlayerType fieldPlayer = board.getFieldPlayer(pos.getX(), pos.getY());
882             if(board.getFieldType(pos.getX(), pos.getY()) == FieldType.TAKEN && fieldPlayer == turn.getCurrentPlayer()){
883                 clickedPlayer = new PlayerPosition(pos.getX(), pos.getY(), fieldPlayer);
884                 
885                 if(clickedPlayer.getPlayerType() == turn.getCurrentPlayer()){
886                     highlightPossibleMoveFields(clickedPlayer);
887                 }
888             }
889             else if(board.getFieldType(pos.getX(), pos.getY()) == FieldType.EMPTY){
890                 if(clickedPlayer!=null){
891                     if(board.isMovePossible(clickedPlayer.getX(), clickedPlayer.getY(), pos.getX(), pos.getY())){
892                         if(board.isClonePossible(clickedPlayer.getX(), clickedPlayer.getY(), pos.getX(), pos.getY())){
893                             board.clone(clickedPlayer.getX(), clickedPlayer.getY(), pos.getX(), pos.getY());
894                             soundManager.play(SoundType.CLONE);
895                         }
896                         if(board.isJumpPossible(clickedPlayer.getX(), clickedPlayer.getY(), pos.getX(), pos.getY())){
897                             board.jump(clickedPlayer.getX(), clickedPlayer.getY(), pos.getX(), pos.getY());
898                             soundManager.play(SoundType.JUMP);
899                         }
900                         
901                         layerStage.clear();
902                         clickedPlayer=null;
903                         turn.nextTurn();
904                         updateStage();
905                     }
906                 }
907             }
908         }
