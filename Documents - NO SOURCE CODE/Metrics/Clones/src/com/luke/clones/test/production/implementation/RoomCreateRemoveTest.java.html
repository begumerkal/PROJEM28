36  public class RoomCreateRemoveTest extends ProductionTest{
37      private final static int S_TO_MS_MULTIPLER=1000;
38      
39      private int roomExistingTimeInSeconds = 30;
40      private int roomNonExistingTimeInSeconds = 10;
41      private int numberOfRooms = 100;
42      private int nextRoomCreationIntervalInSeconds = 1;
43      
44      private List<TestRoomInfoWrapper> testRooms;
45      
46      private TimerTask startTask;
47      private TimerTask testLoopTask;
48      
49      private ClonesClient roomCreateRemoveTestClonesClient;
50  
51      public RoomCreateRemoveTest(AndroidNativeHandler androidNativeHandler,
52              int serverNumberOnServerList) {
53          super(androidNativeHandler, serverNumberOnServerList);
54          super.setNumberOfClients(1);
55      }
56  
57      @Override
58      protected void test() {
59          initialize();
60          prepareAndRunStartTask();
61          prepareAndRunTestLoopTask();
62      }
63      
64      private void initialize() {
65          roomCreateRemoveTestClonesClient = clonesClientTestWrappers.get(0).getClonesClient();
66          testRooms = new ArrayList<TestRoomInfoWrapper>();
67      }
68      
69      private void prepareAndRunStartTask() {
70          startTask = new TimerTask() {
71              @Override
72              public void run() {
73                  startLoop();
74              }
75          };
76          
77          Timer startTimer = new Timer();
78          startTimer.schedule(startTask, 0, nextRoomCreationIntervalInSeconds*S_TO_MS_MULTIPLER);
79      }
80      
81      private void startLoop(){
82          if(testRooms.size()<numberOfRooms){
83              TestRoomInfoWrapper testRoom = new TestRoomInfoWrapper();
84              testRoom.createRoom(roomCreateRemoveTestClonesClient);
85              testRooms.add(testRoom);
86          }
87          else{
88              startTask.cancel();
89          }
90      }
91  
92      private void prepareAndRunTestLoopTask() {
93          testLoopTask = new TimerTask() {
94              @Override
95              public void run() {
96                  testLoop();
97              }
98          };
99          
100         Timer testLoopTimer = new Timer();
101         testLoopTimer.schedule(testLoopTask, 1000, 1000);
102     }
103     
104     private void testLoop(){
105         for(TestRoomInfoWrapper testRoom: testRooms){
106             if(testRoom.getTestRoomState()==TestRoomState.EXISTING){
107                 if(testRoom.getExistingTime().getMeasuredTimeInMs()>roomExistingTimeInSeconds*S_TO_MS_MULTIPLER){
108                     testRoom.removeRoom(roomCreateRemoveTestClonesClient);
109                 }
110             }
111             else if(testRoom.getTestRoomState()==TestRoomState.NON_EXISTING){
112                 if(testRoom.getNonExistingTime().getMeasuredTimeInMs()>roomNonExistingTimeInSeconds*S_TO_MS_MULTIPLER){
113                     testRoom.createRoom(roomCreateRemoveTestClonesClient);
114                 }
115             }
116         }
117     }
118 
119     @Override
120     protected void stopTest() {
121         try {
122             if(startTask!=null) startTask.cancel();
123             if(testLoopTask!=null) testLoopTask.cancel();
124             if(testRooms!=null){
125                 for(TestRoomInfoWrapper testRoom: testRooms){
126                     testRoom.removeRoom(roomCreateRemoveTestClonesClient);
127                     Thread.sleep(100);
128                 }
129             }
130             roomCreateRemoveTestClonesClient.stop();
131         } catch (InterruptedException e) {
132             e.printStackTrace();
133         }
134     }
