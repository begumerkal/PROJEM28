45  public class FlashingTextButton extends TextButton{
46      private static final float DEFAULT_ALPHA = 1f;
47      private static final float DEFAULT_ALPHA_DURATION = 1/17f;
48      
49      List<Float> alphas;
50      int currentAlphaNumber = 0;
51      private float singleAlphaDuration = DEFAULT_ALPHA_DURATION;
52      
53      private float elapsedTime = 0f;
54      private float fieldAlpha = DEFAULT_ALPHA;
55      
56      private boolean flashing = false;
57      
58      List<HighlightActorAnimatedColorChange> highlightActorAnimatedColorChanges;
59      
60      public FlashingTextButton(String text, Skin skin) {
61          super(text, skin);
62      }
63  
64      private FlashingTextButton(String text, Skin skin, String styleName) {
65          super(text, skin, styleName);
66      }
67  
68      private FlashingTextButton(String text, TextButtonStyle style) {
69          super(text, style);
70      }
71      
72  //  @Override
73  //  public void draw(SpriteBatch batch, float parentAlpha) {
74      @Override
75      public void draw(Batch batch, float parentAlpha) {
76          if(flashing){
77              elapsedTime += Gdx.graphics.getDeltaTime();
78              if(elapsedTime>=singleAlphaDuration){
79                  fieldAlpha = choiceNextAlpha();
80                  elapsedTime-=singleAlphaDuration;
81              }
82              super.draw(batch, fieldAlpha);
83          }
84          else{
85              super.draw(batch, parentAlpha);
86          }
87      }
88      
89      private static float choiceStartAlpha(List<Float> alphas){
90          float startAlpha;
91          if(alphas!=null && alphas.size()>0){
92              startAlpha=alphas.get(0);
93          }
94          else{
95              startAlpha = DEFAULT_ALPHA;
96          }
97          return startAlpha;
98      }
99      
100     private float choiceNextAlpha(){
101         currentAlphaNumber++;
102         
103         float nextAlpha;
104         if(alphas==null || alphas.size()==0){
105             return DEFAULT_ALPHA;
106         }
107         if(currentAlphaNumber>alphas.size()-1){
108             currentAlphaNumber=0;
109         }
110         
111         nextAlpha=alphas.get(currentAlphaNumber);
112         
113         return nextAlpha;
114     }
115     
116     public void startFlashing(){
117         fieldAlpha = choiceStartAlpha(alphas);
118         currentAlphaNumber = 0;
119         elapsedTime = 0;
120         flashing = true;
121     }
122     
123     public void stopFlashing(){
124         flashing = false;
125     }
126 
127     public List<Float> getAlphas() {
128         return alphas;
129     }
130 
131     public void setAlphas(List<Float> alphas) {
132         this.alphas = alphas;
133     }
