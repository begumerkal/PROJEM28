37  public abstract class StatusBar{
38      private Stage stage;
39      protected Turn turn;
40      protected LogicBoard logicBoard;
41      private Skin skin, secondSkin;
42      private int x, y, width, height;
43      private Table table;
44      protected Label turnL;
45      protected PlayerType[] availablePlayers;
46      protected Label[] countLabel;
47      
48      public StatusBar(Stage stage, Turn turn, LogicBoard logicBoard,  Skin skin, Skin secondSkin){
49          this.stage = stage;
50          this.turn = turn;
51          this.logicBoard = logicBoard;
52          this.skin = skin;
53          this.secondSkin = secondSkin;
54          
55          table = new Table(skin);
56          table.setBackground(skin.getDrawable("default-round-large"));
57          if(Config.debug) table.debug();
58          
59          stage.addActor(table);
60          
61          Config.log("In status max elment:"+(turn.getAvailablePlayers().length-1));
62          
63          turnL = new Label(turn.getCurrentPlayer()+ " TURN", skin);
64          turnL.setColor(Config.textColor);
65          table.add(turnL);
66          
67          availablePlayers = turn.getAvailablePlayers();
68          countLabel = new Label[availablePlayers.length];
69          if(availablePlayers.length<3){
70              for(int i=0; i<availablePlayers.length; i++){
71                  Image space = new Image(new Texture(Gdx.files.internal("data/skins/space.png")));
72                  table.add(space);
73                  countLabel[i] = new Label(availablePlayers[i]+": "+logicBoard.getBallCount(availablePlayers[i]),secondSkin);
74                  countLabel[i].setColor(Config.textColor);
75                  table.add(countLabel[i]);
76              }
77              Image space = new Image(new Texture(Gdx.files.internal("data/skins/space.png")));
78              table.add(space);
79          }
80          else{
81              Image space = new Image(new Texture(Gdx.files.internal("data/skins/space.png")));
82              table.add(space);
83              Table smallTable = new Table();
84              for(int i=0; i<availablePlayers.length; i++){
85                  if(i>0 && !(i==2&&availablePlayers.length==3)){
86                          space = new Image(new Texture(Gdx.files.internal("data/skins/space.png")));
87                          smallTable.add(space);
88                  }
89                  if(i==2) smallTable.row();
90                  countLabel[i] = new Label(availablePlayers[i]+": "+logicBoard.getBallCount(availablePlayers[i]),secondSkin);
91                  countLabel[i].setColor(Config.textColor);
92                  smallTable.add(countLabel[i]);
93              }
94              space = new Image(new Texture(Gdx.files.internal("data/skins/space.png")));
95              smallTable.add(space);
96              table.add(smallTable);
97              if(availablePlayers.length==3){
98                  space = new Image(new Texture(Gdx.files.internal("data/skins/space.png")));
99                  table.add(space);
100             }
101         }
102         
103         stage.addActor(table);
104     }
105     
106     public void setPosition(int posX, int posY){
107         x = posX;
108         y = posY;
109         table.setPosition(x, y);
110     }
111     
112     public int getPosX(){
113         return x;
114     }
115     
116     public int getPosY(){
117         return y;
118     }
119     
120     public void setSize(int width, int height){
121         this.width = width;
122         this.height = height;
123         table.setSize(width, height);
124     }
125     
126     public int getWidth(){
127         return width;
128     }
129     
130     public int getHeight(){
131         return height;
132     }
133     
134     
135     
136     public LogicBoard getLogicBoard() {
137         return logicBoard;
138     }
139 
140     public void setLogicBoard(LogicBoard logicBoard) {
141         this.logicBoard = logicBoard;
142     }
143 
144     public abstract void reflesh();
145 }
