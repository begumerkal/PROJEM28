35  public class SoundMessage {
36      private final static int IMAGE_SIZE = 96;
37      
38      private Button messageButton;
39      
40      private Skin skin;
41      private String text;
42      
43      public SoundMessage(String text, Skin skin){
44          this.text = text;
45          this.skin = skin;
46          init();
47      }
48      
49      private void init(){
50          setUpButton();
51          messageButton.add(prepareImage()).size(IMAGE_SIZE).space(3);
52          messageButton.add(prepareText()).width(Gdx.graphics.getWidth()-6-IMAGE_SIZE-3);
53      }
54      
55      private void setUpButton(){
56          messageButton = new Button(skin);
57      }
58      
59      private Image prepareImage(){
60          String texturePath = "data/messages/speaker.png";
61          Texture imageTexture = new Texture(Gdx.files.internal(texturePath));
62          imageTexture.setFilter(TextureFilter.Linear, TextureFilter.Linear);
63          Image image = new Image(imageTexture);
64          image.setColor(Config.textColor);
65          return image;
66      }
67      
68      private Label prepareText(){
69          Label textLabel = new Label(text, skin);
70          textLabel.setWrap(true);
71          textLabel.setAlignment(Align.center);
72          return textLabel;
73      }
74  
75      public Button getMessageButton() {
76          return messageButton;
77      }
78  
79      public Skin getSkin() {
80          return skin;
81      }
82  
83      public String getText() {
84          return text;
85      }
86      
87  }
