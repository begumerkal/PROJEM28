334     public void playerMove(MoveBoard playerMove, PlayerType senderPlayerType){
335         PlayerPosition playerPos = playerMove.getPlayerPosition();
336         Position movePos = playerMove.getPosition();
337         Player[] playerArray = getAllPlayers();
338         
339         Config.log("CurrentPlayer:"+turn.getCurrentPlayer()+" Sender:"+senderPlayerType);
340         
341         if(turn.getCurrentPlayer() == senderPlayerType){
342             
343             if(playerMove.getActionType()==ActionTypeMove.MOVE){
344                 PlayerType fieldPlayer = board.getFieldPlayer(playerPos.getX(), playerPos.getY());
345                 if(board.getFieldType(playerPos.getX(), playerPos.getY()) == FieldType.TAKEN && fieldPlayer == turn.getCurrentPlayer()){
346                     if(board.getFieldType(movePos.getX(), movePos.getY()) == FieldType.EMPTY){
347                         if(board.isMovePossible(playerPos.getX(), playerPos.getY(), movePos.getX(), movePos.getY())){
348                             UpdateBoard updateBoard = null;
349                             
350                             if(board.isClonePossible(playerPos.getX(), playerPos.getY(), movePos.getX(), movePos.getY())){
351                                 board.clone(playerPos.getX(), playerPos.getY(), movePos.getX(), movePos.getY());
352                                 updateBoard = new UpdateBoard(ActionTypeBoard.CLONE,playerPos,movePos);
353                             }
354                             else if(board.isJumpPossible(playerPos.getX(), playerPos.getY(), movePos.getX(), movePos.getY())){
355                                 board.jump(playerPos.getX(), playerPos.getY(), movePos.getX(), movePos.getY());
356                                 updateBoard = new UpdateBoard(ActionTypeBoard.JUMP,playerPos,movePos);
357                             }
358                             turn.nextTurn();
359                             //sendBoardUpdateToEveryPlayerInTheRoom
360                             sendToAllPlayers(updateBoard);
361                         }
362                     }
363                 }
364             }
365             else if(playerMove.getActionType()==ActionTypeMove.SKIP_TURN){
366                 skipTurn();
367             }
368             
369         updateStage();
370         }

373     public void updateStage(){
374         //skips player turn when he is eliminated (when he has 0 balls) and when he has no opportunity to move anywhere
375         log.debug("Current player color: "+turn.getCurrentPlayer()+
376                 ";number of his balls:"+board.getBallCount(turn.getCurrentPlayer())+
377                 ";is he moveable?:"+board.isPlayerMoveable(turn.getCurrentPlayer()));
378         
379         if(board.getBallCount(turn.getCurrentPlayer()) == 0 || board.isPlayerMoveable(turn.getCurrentPlayer())==false){
380 
381             //game end condition
382             boolean endGame = true;
383             PlayerType[] availablePlayers = turn.getAvailablePlayers();
384             for(PlayerType player:availablePlayers){
385                 if(player!=turn.getCurrentPlayer()){
386                     if(board.isPlayerMoveable(player)==true) endGame=false;
387                 }
388             }
389             
390             //what should be done at the end
391             //game Ends
392             if(endGame){
393                 pause = true;
394                 playTime.stop();
395                 stopCheckingPlayerMoveTime();
396                 gameRound.nextRound();
397                 
398                 PlayerType[] playerList = turn.getAvailablePlayers();
399                 int[] scoreList = new int[playerList.length];
400                 PlayerScore[] playerScores = new PlayerScore[playerList.length];
401                 for(int i=0; i<scoreList.length;i++){
402                     scoreList[i] = board.getBallCount(playerList[i]);
403                     playerScores[i] = new PlayerScore(playerList[i], scoreList[i]);
404                 }
405                 
406                 Arrays.sort(playerScores, new Comparator<PlayerScore>() {
407                     @Override
408                     public int compare(PlayerScore o1, PlayerScore o2) {
409                         if(o1.getPlayerScore()<o2.getPlayerScore()) return 1;
410                         else if(o1.getPlayerScore()>o2.getPlayerScore()) return -1;
411                         else return 0;
412                     }
413                 });
414                 
415                 boolean firstRound;
416                 if(getTurnStats()==null){
417                     firstRound = true;
418                     try {
419                         setTurnStats(new TurnStats(playerScores));
420                     } catch (TurnStatsArgumentException e) {
421                         e.printStackTrace();
422                     }
423                 }
424                 else{
425                     firstRound = false;
426                     try {
427                         getTurnStats().saveTurnStats(playerScores);
428                     } catch (TurnStatsArgumentException e) {
429                         e.printStackTrace();
430                     }
431                 }
432                 
433                 log.debug("bef addFinishedGameResultToStats(playerScores) PlayerScores: "+playerScores);
434 
435                 addFinishedGameResultToStats(playerScores);
436                 
437                 log.debug("aft addFinishedGameResultToStats(playerScores) PlayerScores: "+playerScores);
438                 
439                 GameOver gameOver = new GameOver(playerScores, getTurnStats(), firstRound);
440                 
441                 log.debug("aft game over create PlayerScores: "+playerScores);
442                 
443                 sendToAllPlayers(gameOver);
444                 
445                 log.debug("aft game over send PlayerScores: "+playerScores);
446                 
447                 measureAndControlNextRoundAutoStart();
448                 
449                 log.debug("aft measureAndControlNextRoundAutoStart()");
450             }
451             
452             //when it's not end but just current player is stuck for a round
453             else{
454                 log.debug("skipping turn for player "+turn.getCurrentPlayer());
455                 skipTurn();
456             }
457             
458         }
459         
460 //      if(board.isPlayerMoveable(turn.getCurrentPlayer())==true){
461 //          log.debug("if(board.isPlayerMoveable(turn.getCurrentPlayer())==true){");
462 //      }
463         
464         if(board.isPlayerMoveable(turn.getCurrentPlayer())==true && AutoGameUtil.isOnlyOnePlayerMoveable(turn, board)){
465             log.debug("if(board.isPlayerMoveable(turn.getCurrentPlayer())==true && isOnlyOnePlayerMoveable()){");
466             List<Position> thisPlayerBallPositions = AutoGameUtil.getThisPlayerBallPositions(turn, board);
467             List<PossibleMove> possibleClones = new ArrayList<PossibleMove>();
468             List<PossibleMove> possibleJumps = new ArrayList<PossibleMove>();
469             possibleClones = AutoGameUtil.getPossibleMoves(BotMoveType.CLONE, thisPlayerBallPositions, board);
470             possibleJumps = AutoGameUtil.getPossibleMoves(BotMoveType.JUMP, thisPlayerBallPositions, board);
471             if(possibleClones.size()>0){
472                 //clone
473                 PlayerPosition playerPos = new PlayerPosition(possibleClones.get(0).getFrom().getX(),
474                         possibleClones.get(0).getFrom().getY(),turn.getCurrentPlayer());
475                 Position movePos = new Position(possibleClones.get(0).getTo().getX(), possibleClones.get(0).getTo().getY());
476                 
477                 board.clone(playerPos.getX(), playerPos.getY(),
478                         movePos.getX(), movePos.getY());
479                 
480                 turn.nextTurn();
481                 
482                 UpdateBoard updateBoard = new UpdateBoard(ActionTypeBoard.CLONE,playerPos,movePos);
483                 sendToAllPlayers(updateBoard);
484             }
485             else if(possibleJumps.size()>0){
486                 //jump
487                 PlayerPosition playerPos = new PlayerPosition(possibleJumps.get(0).getFrom().getX(),
488                         possibleJumps.get(0).getFrom().getY(),turn.getCurrentPlayer());
489                 Position movePos = new Position(possibleJumps.get(0).getTo().getX(), possibleJumps.get(0).getTo().getY());
490                 
491                 board.jump(playerPos.getX(), playerPos.getY(),
492                         movePos.getX(), movePos.getY());
493                 
494                 turn.nextTurn();
495                 
496                 UpdateBoard updateBoard = new UpdateBoard(ActionTypeBoard.JUMP,playerPos,movePos);
497                 sendToAllPlayers(updateBoard);
498             }
499 
500             Timer timer = new Timer();
501             
502             TimerTask timerTask = new TimerTask() {
503                 @Override
504                 public void run() {
505                     updateStage();
506                 }
507             };
508             
509             timer.schedule(timerTask, com.luke.clones.config.Config.autoendOneMoveTimeMs);
510         }
511         
512         measureAndControlPlayerMoveTime();
513         
514     }

532     private void startCheckingPlayerMoveTime(){
533         if(pause==false){
534             checkingPlayerMoveTimeTask = new TimerTask() {
535                 @Override
536                 public void run() {
537                     float playerMoveTimeNumber = playerMoveTime.getMeasuredTimeInMs()/1000f;
538                     log.debug("Time: "+playerMoveTimeNumber);
539                     if(playerMoveTimeNumber>playerMoveTimeLimit.getLimitInSeconds()){
540                         log.debug("Skipping turn because of elapsed time, time: "+playerMoveTimeNumber);
541                         skipTurn();
542                         updateStage();
543                     }
544                 }
545             };
546 
547             checkingPlayerMoveTimeTimer = new Timer(false);
548             checkingPlayerMoveTimeTimer.schedule(checkingPlayerMoveTimeTask,
549                     playerMoveTimeLimit.getLimitInSeconds()*SECONDS_TO_MILLISECONDS_MULTIPLER+ENSURE_MS);
550         }
551     }
552     
553     private void stopCheckingPlayerMoveTime(){
554         if(checkingPlayerMoveTimeTimer!=null){
555             log.debug("checkingPlayerMoveTimeTimer stopped");
556             checkingPlayerMoveTimeTimer.cancel();
557             checkingPlayerMoveTimeTimer.purge();
558         }
559         else{
560             log.debug("checkingPlayerMoveTimeTimer is null so it can't stopped");
561         }
562     }

564     private synchronized void measureAndControlNextRoundAutoStart(){
565         nextRoundAutoStartTimeMeasure.start();
566         nextRoundAutoStartTimer = new Timer(false);
567         TimerTask hideMessageTask = new TimerTask() {
568             @Override
569             public void run() {
570                 if(nextRoundAutoStartTimeMeasure.getMeasuredTimeInMs()>=
571                         com.luke.clones.config.ConfigUtils.getNextRoundAutoStartTimeDependingOnNumberOfPlayers(playerLimit)*1000){
572                     stopCheckingNextRoundAutoStartTime();
573                     if(getNumberOfPlayers()==getPlayerLimit()){
574                         try {
575                             restartGame();
576                         } catch (MapCreatorArgumentException e) {
577                             log.error("", e);
578                         }
579                     }
580                 }
581             }
582         };
583         
584         nextRoundAutoStartTimer.schedule(hideMessageTask,
585                 com.luke.clones.config.ConfigUtils.getNextRoundAutoStartTimeDependingOnNumberOfPlayers(playerLimit)
586                 *SECONDS_TO_MILLISECONDS_MULTIPLER+ENSURE_MS);
587     }

597     public void addPlayer(Player player){
598         List<PlayerType> notUsedPlayerColors = createNotUsedPlayerColorsList();
599         if(notUsedPlayerColors.size()>0){
600             playerMap.put(player, notUsedPlayerColors.get(0));
601             log.info("Player \""+player.getName()+"\" added as "+getPlayerType(player)+" to room "+name);
602         }
603         else{
604             log.info("Player \""+player.getName()+"\" can't be added to room "+name+" reason: all places are taken");
605         }
606     }

608     private List<PlayerType> createNotUsedPlayerColorsList(){
609         List<PlayerType> notUsedPlayerColors = new ArrayList<PlayerType>(Arrays.asList(PlayerOrder.getPlayerOrderArray()));
610         for(PlayerType playerType: playerMap.values()){
611             if(notUsedPlayerColors.contains(playerType)){
612                 notUsedPlayerColors.remove(playerType);
613             }
614         }
615         return notUsedPlayerColors;
616     }

766     private void addFinishedGameResultToStats(PlayerScore[] playerScores){
767         log.debug("addFinishedGameResultToStats start");
768         long playTimeInSeconds = Math.round(Math.ceil(playTime.getMeasuredTimeInMs()/1000f));
769         //clean play time there
770         for(int i=0; i<playerScores.length;i++){
771             Player tPlayer = MapUtil.getKeyByValue(playerMap, playerScores[i].getPlayerType());
772             log.debug("player: "+tPlayer);
773             String playerGoogleId = tPlayer.getGoogleUserData().getId();
774             
775             log.debug("player id: "+playerGoogleId);
776             
777             //remember gameType
778             GameType gameType = StatsUtil.maxNumberOfPlayersToGameType(playerLimit);
779             
780             WinLoseType winLose;
781             if(i==0){
782                 winLose=WinLoseType.WIN;
783             }
784             else{
785                 winLose=WinLoseType.LOSE;
786             }
787             
788             int ballsScored = playerScores[i].getPlayerScore();
789             
790             GameResult gameResult = new GameResult();
791             gameResult.setWinLose(winLose);
792             gameResult.setBallsScored(ballsScored);
793             gameResult.setPlayTimeInSeconds(playTimeInSeconds);
794             
795             statsAddService.addFinishedGameResult(playerGoogleId, gameType, gameResult);
796         }
797         log.debug("addFinishedGameResultToStats end");
798     }

800     private void addSuddenlyInterruptedGameResultToStats(Player removedPlayer){
801         Set<Player> tPlayers = new HashSet<Player>(playerMap.keySet());
802         if(removedPlayer!=null) tPlayers.add(removedPlayer);
803         long playTimeInSeconds = Math.round(Math.ceil(playTime.getMeasuredTimeInMs()/1000f));
804         //clean playTime there
805         for(Player tPlayer: tPlayers){
806             String playerGoogleId = tPlayer.getGoogleUserData().getId();
807             GameType gameType = StatsUtil.maxNumberOfPlayersToGameType(playerLimit);
808             SuddenlyInterruptedGameResult suddenlyInterruptedGameResult = new SuddenlyInterruptedGameResult();
809             suddenlyInterruptedGameResult.setPlayTimeInSeconds(playTimeInSeconds);
810             statsAddService.addSuddenlyInterruptedGameResult(playerGoogleId, gameType, suddenlyInterruptedGameResult);
811         }
812     }
