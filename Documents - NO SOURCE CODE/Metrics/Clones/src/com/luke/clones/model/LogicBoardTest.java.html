44          board.setFieldPlayer(board.getWidth()-1, board.getHeight()-1, PlayerType.RED);
45          
46          boolean testB;
47          
48          testB = board.isMovePossible(2, 3, 1, 2);
49          Assert.assertEquals(true, testB);
50          testB = board.isMovePossible(2, 3, 1, 4);
51          Assert.assertEquals(true, testB);
52          testB = board.isMovePossible(2, 3, 1, 3);
53          Assert.assertEquals(true, testB);
54          
55          testB = board.isClonePossible(2, 3, 3, 2);
56          Assert.assertEquals(true, testB);
57          testB = board.isClonePossible(2, 3, 3, 3);
58          Assert.assertEquals(true, testB);
59          testB = board.isClonePossible(2, 3, 3, 4);
60          Assert.assertEquals(true, testB);
61          testB = board.isClonePossible(2, 3, 2, 2);
62          Assert.assertEquals(true, testB);
63          testB = board.isClonePossible(2, 3, 1, 2);
64          Assert.assertEquals(true, testB);
65          testB = board.isClonePossible(2, 3, 2, 4);
66          Assert.assertEquals(true, testB);
67          testB = board.isClonePossible(2, 3, 1, 4);
68          Assert.assertEquals(true, testB);
69          testB = board.isClonePossible(2, 3, 1, 3);
70          Assert.assertEquals(true, testB);
71          
72          testB = board.isJumpPossible(2, 3, 5, 4);
73          Assert.assertEquals(false, testB);
