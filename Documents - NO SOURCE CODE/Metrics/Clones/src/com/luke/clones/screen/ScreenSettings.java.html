46  public class ScreenSettings implements Screen{
47      private StartScreen startScreen;
48      private int width, height;
49      private Viewport viewport;
50      private Stage stage;
51      private Skin skin;
52      private Table mainTable;
53      
54      private InputMultiplexer inputMultiplexer;
55      private SpecialKeyHandler specialKeyHandler;
56      
57      public ScreenSettings(StartScreen startScreen){
58          this.startScreen = startScreen;
59      }
60  
61      @Override
62      public synchronized void show() {
63          width = Config.width;
64          height = Config.height;
65  //      stage = new Stage(width, height, false);
66          viewport = new FitViewport(width, height);
67          stage = new Stage();
68          stage.setViewport(viewport);
69          
70          specialKeyHandler = new SpecialKeyHandler() {
71              @Override
72              public void backKeyAction() {
73                  startScreen.setToMainMenu();
74              }
75          };
76          
77          inputMultiplexer = new InputMultiplexer(stage, specialKeyHandler);
78          
79          Gdx.input.setInputProcessor(inputMultiplexer);
80          skin = new Skin(Gdx.files.internal("data/skins/skin_medium.json"));
81          
82          loadSettingsFromMemory();
83          
84          buildInterface();
85      }
86  
87      @Override
88      public synchronized void hide() {
89          stage = null;
90          skin = null;
91      }
92  
93      @Override
94      public synchronized void render(float delta) {
95          Gdx.gl.glClearColor(54f/255f, 47f/255f, 45f/255f, 1);
96  //      Gdx.gl.glClear(GL10.GL_COLOR_BUFFER_BIT);
97          Gdx.gl.glClear(GL20.GL_COLOR_BUFFER_BIT);
98          
99          stage.act();
100         stage.draw();
101     }
102     
103     private void loadSettingsFromMemory() {
104 
105     }
106     
107     private void buildInterface() {
108         buildMainTable();
109         
110         TextButton backToMenuButton = new TextButton("Back to menu", skin);
111         mainTable.add(backToMenuButton).minSize(width*3/4-6, 70).spaceBottom(3).spaceTop(3);
112         
113         mainTable.row();
114         
115         backToMenuButton.addListener(new ClickListener(){
116             @Override
117             public void clicked(InputEvent event, float x, float y) {
118                 SoundManager.INSTANCE.play(SoundType.CLICK);
119                 startScreen.setToMainMenu();
120             }
121         });
122         
123         CheckBox musicBox = new CheckBox("Music", skin);
124         musicBox.setChecked(SettingsManager.getSettingValue(SettingType.MUSIC_ON));
125         mainTable.add(musicBox).minHeight(40).spaceTop(12);
126         
127         mainTable.row();
128         
129         CheckBox soundsBox = new CheckBox("Sounds", skin);
130         soundsBox.setChecked(SettingsManager.getSettingValue(SettingType.SOUND_ON));
131         mainTable.add(soundsBox).minHeight(40).spaceTop(5);
132         
133         musicBox.addListener(new ClickListener(){
134             @Override
135             public void clicked(InputEvent event, float x, float y) {
136                 SoundManager.INSTANCE.play(SoundType.CLICK);
137                 reverseSettingValueAndSaveIt(SettingType.MUSIC_ON);
138             }
139         });
140         
141         soundsBox.addListener(new ClickListener(){
142             @Override
143             public void clicked(InputEvent event, float x, float y) {
144                 SoundManager.INSTANCE.play(SoundType.CLICK);
145                 reverseSettingValueAndSaveIt(SettingType.SOUND_ON);
146             }
147         });
148     }
149     
150     private void reverseSettingValueAndSaveIt(SettingType setting){
151         boolean currentValue = SettingsManager.getSettingValue(setting);
152         boolean newValue = !currentValue;
153         SettingsManager.setSettingValue(setting, newValue);
154         
155         updateMusicPlayStateDependingOnSettings();
156     }
157     
158     private void updateMusicPlayStateDependingOnSettings(){
159         if(SettingsManager.getSettingValue(SettingType.MUSIC_ON)==true){
160             startScreen.getMusicManager().play();
161         }
162         else{
163             startScreen.getMusicManager().pauseAndScheduleGradualVolumeIncreaseBeforePlay();
164         }
165     }
166     
167     private void buildMainTable(){
168         mainTable = new Table(skin);
169         mainTable.setPosition(0+Config.widthOffset/2, Config.menuBottomHeightOffset);
170         mainTable.setSize(width-Config.widthOffset/2, height-(Config.menuBottomHeightOffset+Config.menuTopHeightOffset));
171         mainTable.setBackground(skin.getDrawable("default-scroll"));
172         mainTable.align(Align.top);
173         
174         stage.addActor(mainTable);
175     }
