38  public class TestRoomInfoWrapper {
39      private RoomCreateInfo roomCreateInfo;
40      private TestRoomState testRoomState;
41      private TimeMeasure existingTime;
42      private TimeMeasure nonExistingTime;
43      
44      public TestRoomInfoWrapper(){
45          existingTime = new TimeMeasure();
46          nonExistingTime = new TimeMeasure();
47      }
48      
49      public void createRoom(ClonesClient clonesClient){
50          System.out.println("createRoom");
51          
52          String roomName = prepareRoomInfo();
53          
54          clonesClient.send(new MoveRoom(ActionTypeMoveRoom.CREATE, roomCreateInfo, roomName));
55          testRoomState = TestRoomState.EXISTING;
56          existingTime.start();
57      }
58      
59      public void createRoom(ClonesClient clonesClient, RoomCreateInfo externalRoomCreateInfo){
60          System.out.println("createRoom");
61          
62          String roomName = prepareRoomInfo();
63          
64          clonesClient.send(new MoveRoom(ActionTypeMoveRoom.CREATE, externalRoomCreateInfo, roomName));
65          testRoomState = TestRoomState.EXISTING;
66          existingTime.start();
67      }
68  
69      private String prepareRoomInfo() {
70          String roomName = "Test room "+RandomStringUtils.randomAlphanumeric(10);
71          int playerNumber = 2;
72          PlayerMoveTimeLimit playerMoveTimeLimit = PlayerMoveTimeLimitType.T15S.getPlayerMoveTimeLimit();
73          String mapName = AvailableMapModelsType.TUNNEL_RATS.getMap().getMapName();
74          
75          roomCreateInfo = new RoomCreateInfo(
76                  roomName,
77                  playerNumber,
78                  playerMoveTimeLimit,
79                  mapName);
80          return roomName;
81      }
82      
83      public void removeRoom(ClonesClient clonesClient){
84          System.out.println("removeRoom");
85          clonesClient.send(new MoveRoom(ActionTypeMoveRoom.LEAVE, null, null));
86          testRoomState = TestRoomState.NON_EXISTING;
87          nonExistingTime.start();
88      }
89      
90      public void joinRoom(ClonesClient clonesClient){
91          clonesClient.send(new MoveRoom(ActionTypeMoveRoom.ENTER, null, roomCreateInfo.getRoomName()));
92      }
93      
94      public void skipTurn(ClonesClient clonesClient){
95          clonesClient.send(new MoveBoard(ActionTypeMove.SKIP_TURN, null, null));
96      }
97      
98      public void startGame(ClonesClient clonesClient){
99          clonesClient.send(new MoveRoom(ActionTypeMoveRoom.START_GAME, null, null));
100     }
101 
102     public TestRoomState getTestRoomState() {
103         return testRoomState;
104     }
105 
106     public TimeMeasure getExistingTime() {
107         return existingTime;
108     }
109 
110     public TimeMeasure getNonExistingTime() {
111         return nonExistingTime;
112     }
113 
114     public RoomCreateInfo getRoomCreateInfo() {
115         return roomCreateInfo;
116     }

118 }
