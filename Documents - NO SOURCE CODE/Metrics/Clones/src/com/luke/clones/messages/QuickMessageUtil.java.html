34  public class QuickMessageUtil {
35      public static List<QuickMessage> prepareQuickMessagesWithNewAchievements(AchievementType[] newAchievements, Skin skin, Skin smallSkin, StartScreen startScreen){
36          List<QuickMessage> quickMessagesAboutAchievements = new ArrayList<QuickMessage>();
37          for(AchievementType achievement: newAchievements){
38              QuickMessage quickMessageAboutAchievement = new QuickMessage(achievement.getTitle(), skin, smallSkin);
39              quickMessageAboutAchievement.setOptionalTitle("Achievement Unlocked");
40              quickMessageAboutAchievement.setOptionalAdditionalText("["+achievement.getDescription()+"]");
41              QuickMessageType quickMessageType = convertAchievementGenreTypeToQuickMessageType(achievement);
42              if(quickMessageType!=null) quickMessageAboutAchievement.setQuickMessageType(quickMessageType);
43              quickMessageAboutAchievement.setAutoHide(false);
44              if(achievement.isShowRateItButton()){
45                  quickMessageAboutAchievement.setShowRateContent(true, startScreen);
46              }
47              quickMessageAboutAchievement.setConfirmButtonText("Close");
48              quickMessagesAboutAchievements.add(quickMessageAboutAchievement);
49          }
50          Collections.reverse(quickMessagesAboutAchievements);
51          return quickMessagesAboutAchievements;
52      }
53      
54      private static QuickMessageType convertAchievementGenreTypeToQuickMessageType(AchievementType achievementType){
55          if(achievementType.getAchievementGenreType()==AchievementGenreType.GAMES_PLAYED){
56              return QuickMessageType.ACHIEVEMENT_GAMES_PLAYED;
57          }
58          else if(achievementType.getAchievementGenreType()==AchievementGenreType.GAMES_WON){
59              return QuickMessageType.ACHIEVEMENT_GAMES_WON;
60          }
61          else if(achievementType.getAchievementGenreType()==AchievementGenreType.GAMES_WON_IN_ROW){
62              return QuickMessageType.ACHIEVEMENT_GAMES_WON_IN_ROW;
63          }
64          else if(achievementType.getAchievementGenreType()==AchievementGenreType.PLAY_TIME){
65              return QuickMessageType.ACHIEVEMENT_PLAY_TIME;
66          }
67          return null;
68      }
