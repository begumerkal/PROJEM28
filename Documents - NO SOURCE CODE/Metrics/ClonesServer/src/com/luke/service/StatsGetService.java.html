38  public class StatsGetService {
39      public static StatsGetService INSTANCE = new StatsGetService();
40      
41      private Logger log = Logger.getLogger(StatsGetService.class);
42      
43      private StatsDAO statsDAO = StatsDAO.INSTANCE;
44      private UsersDAO usersDAO = UsersDAO.INSTANCE;
45      
46      private StatsGetService(){
47          
48      }
49      
50      public long getGeneralStatValue(String userGoogleId, StatsType statsType){
51          UserDTO userDTO = usersDAO.getUserByGoogleId(userGoogleId);
52          GeneralStatsTypeDTO statisticTypeDTO = statsDAO.getStatisticTypeByStatsType(statsType);
53          GeneralStatsDTO generalStatsDTO = statsDAO.getGeneralStatsByUserAndGeneralStatsType(userDTO.getId(), statisticTypeDTO.getId());
54          if(generalStatsDTO!=null){
55              return generalStatsDTO.getValue();
56          }
57          else{
58              return 0;
59          }
60      }
61      
62      public Map<StatsType, Long> getAllGeneralStatsValuesMap(String userGoogleId){
63          StatsType[] generalStatsTypes = StatsType.values();
64          Map<StatsType, Long> generalStatsMap = new HashMap<StatsType, Long>();
65          
66          for(StatsType statsType: generalStatsTypes){
67              long value = getGeneralStatValue(userGoogleId, statsType);
68              generalStatsMap.put(statsType, value);
69          }
70          return generalStatsMap;
71      }
72      
73      public long getUserLevel(String userGoogleId){
74          long  userPoints = getUserPoints(userGoogleId);
75          LevelPointsDTO currentLevelPointsDTO = statsDAO.getLevelPointsByPoints(userPoints);
76          long userLevel = currentLevelPointsDTO.getLevel();
77          return userLevel;   
78      }
79      
80      public long getUserPercentToNextLevel(String userGoogleId){
81          long  userPoints = getUserPoints(userGoogleId);
82          LevelPointsDTO currentLevelPointsDTO = statsDAO.getLevelPointsByPoints(userPoints);
83          long userLevel = currentLevelPointsDTO.getLevel();
84          long currentLevelPoints = currentLevelPointsDTO.getPoints();
85  
86          LevelPointsDTO nextLevelPointsDTO = statsDAO.getLevelPointsByLevel(userLevel+1);
87          long nextLevelPoints = nextLevelPointsDTO.getPoints();
88          
89          long pointsDifferenceBetweenLevels = nextLevelPoints-currentLevelPoints;
90          long userPointsAboveCurrentLevel = userPoints-currentLevelPoints;
91          
92          double userPointsAboveCurrentLevelToDifferenceBetweenLevels = (double)userPointsAboveCurrentLevel/(double)pointsDifferenceBetweenLevels;
93          long userPointsAboveCurrentLevelToDifferenceBetweenLevelsPercent =
94                  Math.round(userPointsAboveCurrentLevelToDifferenceBetweenLevels*100);
95          
96          long toNextLevelPercent = 100 - userPointsAboveCurrentLevelToDifferenceBetweenLevelsPercent;
97          
98          return toNextLevelPercent;
99      }
100     
101     private long getUserPoints(String userGoogleId){
102         long  userPoints = 0;
103         userPoints+=getGeneralStatValue(userGoogleId, StatsType.WON_2P_GAMES)*100;
104         userPoints+=getGeneralStatValue(userGoogleId, StatsType.WON_4P_GAMES)*150;
105         userPoints+=getGeneralStatValue(userGoogleId, StatsType.TOTAL_BALLS_SCORED);
106         return userPoints;
107     }
