85      private void init(){
86          messageTable = new Table(skin);
87          messageTable.setBackground(skin.getDrawable("message-round-large-dark"));
88          messageTable.setSize(Gdx.graphics.getWidth(), 150);
89          messageTable.setPosition(Gdx.graphics.getWidth()/2-messageTable.getWidth()/2, Gdx.graphics.getHeight()+messageTable.getHeight());
90          
91          timeMeasure = new TimeMeasure();
92      }

94      public synchronized void prepareToShow(){
95          prepareAutoHide();
96          prepareContent();
97          inAnimation();
98      }

100     private synchronized void prepareAutoHide(){
101         if(autoHide){
102             timeMeasure.start();
103             
104             messageTimer = new Timer(true);
105 
106             TimerTask hideMessageTask = new TimerTask() {
107                 @Override
108                 public void run() {
109                     Gdx.app.postRunnable(new Runnable() {
110                         public void run() {
111                             updateTimeText();
112                         }
113                     });
114                     if(timeMeasure.getMeasuredTimeInMs()>=autoHideTimeInSecods*1000){
115                         Gdx.app.postRunnable(new Runnable() {
116                             public void run() {
117                                 hideMessage();
118                             }
119                         });
120                     }
121                 }
122             };
123             
124             messageTimer.scheduleAtFixedRate(hideMessageTask, 0, 1000);
125         }
126     }

128     private synchronized void prepareContent(){
129         messageTable.clear();
130         messageTable.align(Align.top);
131 //      messageTable.debug();
132         
133         Label spaceLabel = new Label("_", skin);
134         
135         Table twoColumnTable = new Table();
136         twoColumnTable.setWidth(messageTable.getWidth());
137         Table leftColumn = new Table();
138         leftColumn.setWidth(IMAGE_SIZE);
139         Table rightColumn = new Table();
140         rightColumn.setWidth(twoColumnTable.getWidth()-IMAGE_SIZE);
141         
142         leftColumn.add(prepareImage()).size(IMAGE_SIZE, IMAGE_SIZE).left().top();
143         
144         if(optionalTitle.length()>0){
145             rightColumn.add(prepareOptionalTitle()).width(twoColumnTable.getWidth()-IMAGE_SIZE).spaceBottom(7);
146             rightColumn.row();
147         }
148         rightColumn.add(prepareText()).width(twoColumnTable.getWidth()-IMAGE_SIZE);
149         rightColumn.row();
150         if(optionalAdditionalText.length()>0){
151             rightColumn.add(prepareOptionalAdditionalText()).width(twoColumnTable.getWidth()-IMAGE_SIZE);
152             rightColumn.row();
153         }
154         
155         twoColumnTable.add(leftColumn);
156         twoColumnTable.add(rightColumn);
157         messageTable.add(twoColumnTable);
158         
159         messageTable.row();
160         
161         if(showRateContent){
162             messageTable.setHeight(messageTable.getHeight()+60);
163             messageTable.add(prepareRateContent()).spaceTop(15).spaceBottom(10);
164             messageTable.row();
165         }
166         
167         if(autoHide){
168             messageTable.setHeight(messageTable.getHeight()+spaceLabel.getHeight());
169             messageTable.add(prepareTimeText()).colspan(2);
170             messageTable.row();
171         }
172         if(showConfirmButton){
173             messageTable.add(prepareConfirmButton()).colspan(2).spaceTop(5).spaceBottom(3).minSize(Config.width*1/5-6, 35);
174             messageTable.row();
175         }
176         
177     }

179     private Image prepareImage(){
180         String texturePath = "data/messages/info.png";
181         if(quickMessageType==QuickMessageType.INFO){
182             texturePath = "data/messages/info.png";
183         }
184         else if(quickMessageType==QuickMessageType.ALERT){
185             texturePath = "data/messages/alert.png";
186         }
187         else if(quickMessageType==QuickMessageType.ACHIEVEMENT_GAMES_WON){
188             texturePath = "data/messages/achievement_won_games.png";
189         }
190         else if(quickMessageType==QuickMessageType.ACHIEVEMENT_GAMES_PLAYED){
191             texturePath = "data/messages/achievement_played_games.png";
192         }
193         else if(quickMessageType==QuickMessageType.ACHIEVEMENT_PLAY_TIME){
194             texturePath = "data/messages/achievement_time.png";
195         }
196         else if(quickMessageType==QuickMessageType.ACHIEVEMENT_GAMES_WON_IN_ROW){
197             texturePath = "data/messages/achievement_won_in_row.png";
198         }
199         
200         Texture imageTexture = new Texture(Gdx.files.internal(texturePath));
201         imageTexture.setFilter(TextureFilter.Linear, TextureFilter.Linear);
202         Image image = new Image(imageTexture);
203         image.setColor(Config.textColor);
204         return image;
205     }

228     private Table prepareRateContent(){
229         Label textLabel = new Label("If you have a moment please rate Clones on the Play Store", smallSkin);
230         textLabel.setWrap(true);
231         textLabel.setAlignment(Align.center);
232         
233         //179x47
234         Texture clonesPlayLogoTexture = new Texture(Gdx.files.internal("data/clonesplaylogo.png"));
235         clonesPlayLogoTexture.setFilter(TextureFilter.Linear, TextureFilter.Linear);
236         Texture clonesPlayLogoAltTexture = new Texture(Gdx.files.internal("data/clonesplaylogoalt.png"));
237         clonesPlayLogoAltTexture.setFilter(TextureFilter.Linear, TextureFilter.Linear);
238         final TextureRegion clonesPlayLogoRegion = new TextureRegion(clonesPlayLogoTexture, 0, 94, 256, 161-94);
239         final TextureRegion clonesPlayLogoAltRegion = new TextureRegion(clonesPlayLogoAltTexture, 0, 94, 256, 161-94);
240         Image rateImage = new Image(clonesPlayLogoRegion);
241         rateImage.setAlign(Align.center);
242         
243         rateImage.addListener(new ClickListener(){
244             @Override
245             public boolean touchDown(InputEvent event, float x, float y,
246                     int pointer, int button) {
247                 ((Image) event.getTarget()).setDrawable(new TextureRegionDrawable(clonesPlayLogoAltRegion));
248                 return true;
249             }
250             @Override
251             public void touchUp(InputEvent event, float x, float y,
252                     int pointer, int button) {
253                 ((Image) event.getTarget()).setDrawable(new TextureRegionDrawable(clonesPlayLogoRegion));
254                 startScreen.getAndroidNativeHandler().openMarketApp(Config.appPackageForGooglePlayButton);
255             }
256         });
257         
258         Table table = new Table();
259         table.add(textLabel).width(messageTable.getWidth()/2).minHeight(47);
260         
261         Table tempTable = new Table();
262         tempTable.add(rateImage).maxWidth(179).maxHeight(47);
263         table.add(tempTable).width(messageTable.getWidth()/2).minHeight(47);
264         
265         return table;
266     }

280             timeTextLabel = new Label(""+autoHideTimeInSecods+"s", skin);
281             return timeTextLabel;
282     }

284     private synchronized void inAnimation(){
285         messageTable.addAction(Actions.moveTo(Gdx.graphics.getWidth()/2-messageTable.getWidth()/2, Gdx.graphics.getHeight()/2-messageTable.getHeight()/2, ANIMATION_DURATION));
286     }
287     
288     private synchronized void outAnimation(){
289         SequenceAction sequenceAction = new SequenceAction();
290         sequenceAction.addAction(Actions.moveTo(Gdx.graphics.getWidth()/2-messageTable.getWidth()/2, 0-messageTable.getHeight(), ANIMATION_DURATION));
291         sequenceAction.addAction(new HideMessageAction());
292         messageTable.addAction(sequenceAction);
293     }
294     
295     private class HideMessageAction extends Action{
296         @Override
297         public boolean act(float delta) {
298             messageTable.remove();
299             return true;
300         }
301     }
302     
303     private void hideMessage(){
304         if(autoHide){
305             messageTimer.cancel();
306         }
307         outAnimation();
308     }
309     
310     private void updateTimeText(){
311         if(autoHide){
312             int measuredTimeInSeconds = Math.round(timeMeasure.getMeasuredTimeInMs()/1000f);
313             int timeToClose = autoHideTimeInSecods - measuredTimeInSeconds;
314             timeTextLabel.setText(timeToClose+"s");
315         }
316     }
