25  public class DrawFieldButtonTypeChanger {
26      private static int i = 0;
27      
28      public static DrawFieldButtonType getCurrent(){
29          DrawFieldButtonType drawFieldButtonType = DrawFieldButtonType.values()[i];
30          return drawFieldButtonType;
31      }
32      
33      public static DrawFieldButtonType getNext(){
34          increment();
35          DrawFieldButtonType drawFieldButtonType = DrawFieldButtonType.values()[i];
36          return drawFieldButtonType;
37      }
38      
39      private static void increment(){
40          i++;
41          if(i>=DrawFieldButtonType.values().length){
42              i=0;
43          }
44      }
45      
46  }
