52      public AchievementTypeDTO getAchievementTypeByAchievementType(AchievementType achievementType){
53          try{
54              connection = databaseService.connectToDatabaseIfNotConnectedAndReturnConnection();
55              
56              String selectAchievementTypeStatement = "select id, type from achievements_type where type=?";
57              PreparedStatement preparedStatement = connection.prepareStatement(selectAchievementTypeStatement);
58              preparedStatement.setString(1, achievementType.name());
59              ResultSet resultSet = preparedStatement.executeQuery();
60              while(resultSet.next()){
61                  Long resultId = resultSet.getLong("id");
62                  String resultType = resultSet.getString("type");
63                  return new AchievementTypeDTO(resultId, achievementType);
64              }
65              return null;
66          } catch (SQLException e) {
67              log.error("An error error occured during sql statement execution");
68              log.error("",e);
69              return null;
70          }
71      }

100     public List<AchievementTypeDTO> getUserAchievementTypes(long userId){
101         try{
102             connection = databaseService.connectToDatabaseIfNotConnectedAndReturnConnection();
103             
104             String selectAchievementsByUserAndAchievementsTypeStatement = "select at.id, at.type from achievements a inner join achievements_type at on a.achievements_type=at.id where _user=?";
105             PreparedStatement preparedStatement = connection.prepareStatement(selectAchievementsByUserAndAchievementsTypeStatement);
106             preparedStatement.setLong(1, userId);
107             ResultSet resultSet = preparedStatement.executeQuery();
108             
109             List<AchievementTypeDTO> userAchievementTypes = new ArrayList<AchievementTypeDTO>();
110             while(resultSet.next()){
111                 Long resultId = resultSet.getLong("id");
112                 String resultAchievementType = resultSet.getString("type");
113                 
114                 AchievementTypeDTO achievementTypeDTO = new AchievementTypeDTO(resultId, AchievementType.valueOf(resultAchievementType));
115                 userAchievementTypes.add(achievementTypeDTO);
116             }
117             return userAchievementTypes;
118         } catch (SQLException e) {
119             log.error("An error error occured during sql statement execution");
120             log.error("",e);
121             return new ArrayList<AchievementTypeDTO>();
122         }
123     }

125     public AchievementDTO getAchievementByUserAndAchievementsType(long userId, long achievementTypeId){
126         try{
127             connection = databaseService.connectToDatabaseIfNotConnectedAndReturnConnection();
128             
129             String selectAchievementsByUserAndAchievementsTypeStatement = "select id, achievements_type, _user from achievements where _user=? and achievements_type=?";
130             PreparedStatement preparedStatement = connection.prepareStatement(selectAchievementsByUserAndAchievementsTypeStatement);
131             preparedStatement.setLong(1, userId);
132             preparedStatement.setLong(2, achievementTypeId);
133             ResultSet resultSet = preparedStatement.executeQuery();
134             
135             while(resultSet.next()){
136                 Long resultId = resultSet.getLong("id");
137                 Long resultAchievementType = resultSet.getLong("achievements_type");
138                 Long resultUser = resultSet.getLong("_user");
139                 
140                 AchievementDTO achievementDTO = new AchievementDTO(resultId, resultAchievementType, resultUser);
141                 return achievementDTO;
142             }
143             return null;
144         } catch (SQLException e) {
145             log.error("An error error occured during sql statement execution");
146             log.error("",e);
147             return null;
148         }

151     public void insertAchievementValue(AchievementDTO achievement){
152         try{
153             connection = databaseService.connectToDatabaseIfNotConnectedAndReturnConnection();
154             
155             String insertStatisticStatement = "insert into achievements (achievements_type, _user) values (?,?)";
156             PreparedStatement preparedStatement = connection.prepareStatement(insertStatisticStatement);
157             preparedStatement.setLong(1, achievement.getAchievementType());
158             preparedStatement.setLong(2, achievement.getUser());
159             preparedStatement.executeUpdate();
160         } catch (SQLException e) {
161             log.error("An error error occured during sql statement execution");
162             log.error("",e);
163         }
