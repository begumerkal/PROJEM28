34   public class MapAutoCreator {
35      /**
36       * Creates map that is necessary to start clones game (precisely: to set ScreenLocalMultiplayerGame 
37       * or any other game screen as a current screen or to manually, using game method start game one more time);
38       * 
39       * @param PlayerNumber, accepted: 2, 4
40       * @param mapSize, accepted: "Small", "Medium", "Big"
41       * @param mapShape, accepted: "Rectangle", "Cross"
42       * @param mapObstacles, accepted: "None", "Small Obstacles", "Medium Obstacles", "Big Obstacles"
43       * @return Map containing it's width, height, solids (shape and obstacles) and player positions
44       */
45      public static Map createMap(MapModel mapModel) throws MapCreatorArgumentException {
46          int width; // wide side
47          int height; // narrow side
48          Position[] startSolids; //obstacle fields
49          PlayerPosition[] startPlayers; // where player clones will be at the start
50          MapType mapType;
51          
52          int[] intSize = mapModel.getMapSize();
53          width = intSize[0];
54          height = intSize[1];
55          
56          startSolids = mapModel.getSolids();
57          startPlayers = mapModel.getPlayerPositions();
58          
59          mapType = mapModel.getMapType();
60          
61          return new Map(width, height, startSolids, startPlayers, mapType);
62      }
63      
64      public static Map createMap(AvailableMapModelsType availableMapModelsType) throws MapCreatorArgumentException {
65          return createMap(availableMapModelsType.getMap());
66      }
67      
68      public static class MapCreatorArgumentException extends Exception{
69          private static final long serialVersionUID = 7795742930373657903L;
70  
71          @Override
72          public String getMessage() {
73              String message = "Illegal argument passed to the method";
74              return super.getMessage() + message;
75          }
76      }
