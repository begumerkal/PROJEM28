30  public class LogicBoard {
31      private int width, height;
32      private LogicField[][] field;
33      private HashMap<PlayerType, Integer> ballCount;
34      private int maxBallCount;
35      
36      public LogicBoard(int width, int height){
37          this.width = width;
38          this.height = height;
39          
40          field = new LogicField[width][height];
41          for(int i=0;i<width;i++){
42              for(int j=0; j<height;j++){
43                  field[i][j] = new LogicField(FieldType.EMPTY, PlayerType.NONE);
44              }
45          }
46          
47          ballCount = new HashMap<PlayerType, Integer>();
48          ballCount.put(PlayerType.RED,0);
49          ballCount.put(PlayerType.GREEN,0);
50          ballCount.put(PlayerType.BLUE,0);
51          ballCount.put(PlayerType.ORANGE,0);
52          
53          maxBallCount = width * height;
54      }
55      
56      public void setFieldTypeSolid(int posX, int posY){
57          if(posX>=0 && posY>=0 && posX<width && posY<height){
58                  if(field[posX][posY].getFieldType()== FieldType.TAKEN){
59                      ballCount.put(field[posX][posY].getPlayer(), ballCount.get(field[posX][posY].getPlayer())-1);
60                  }
61                  field[posX][posY].setType(FieldType.SOLID);
62                  field[posX][posY].setPlayer(PlayerType.NONE);
63          }
64          else throw new ArrayIndexOutOfBoundsException();
65      }
66      
67      public void setFieldTypeEmpty(int posX, int posY){
68          if(posX>=0 && posY>=0 && posX<width && posY<height){
69              if(field[posX][posY].getFieldType()!=FieldType.SOLID){
70                  if(field[posX][posY].getFieldType()== FieldType.TAKEN){
71                      ballCount.put(field[posX][posY].getPlayer(), ballCount.get(field[posX][posY].getPlayer())-1);
72                  }
73                  field[posX][posY].setType(FieldType.EMPTY);
74                  field[posX][posY].setPlayer(PlayerType.NONE);
75              }
76          }
77          else throw new ArrayIndexOutOfBoundsException();
78      }
79      
80      public FieldType getFieldType(int posX, int posY){
81          if(posX>=0 && posY>=0 && posX<width && posY<height){
82              return field[posX][posY].getFieldType();
83          }
84          else throw new ArrayIndexOutOfBoundsException();
85      }
86      
87      public void setFieldPlayer(int posX, int posY, PlayerType playerType){
88          if(posX>=0 && posY>=0 && posX<width && posY<height){
89              if(field[posX][posY].getFieldType()!=FieldType.SOLID){
90                  field[posX][posY].setPlayer(playerType);
91                  field[posX][posY].setType(FieldType.TAKEN);
92                  ballCount.put(playerType, ballCount.get(playerType)+1);
93              }
94          }
95          else throw new ArrayIndexOutOfBoundsException();
96      }
97      
98      public PlayerType getFieldPlayer(int posX, int posY){
99          if(posX>=0 && posY>=0 && posX<width && posY<height){
100             return field[posX][posY].getPlayer();
101         }
102         else throw new ArrayIndexOutOfBoundsException();
103     }

105     public boolean isMovePossible(int fromX,int fromY,int toX,int toY){
106         
107         if(fromX == toX && fromY == toY) return false;
108         
109         if(fromX<0 || fromY<0 || fromX>width-1 || fromY>height-1) return false;
110         
111         if(field[fromX][fromY].getFieldType()!=FieldType.TAKEN) return false;
112         if(field[fromX][fromY].getPlayer()==PlayerType.NONE) return false;
113         
114         if(toX<0 || toY<0 || toX>width-1 || toY>height-1) return false;
115         
116         if(field[toX][toY].getFieldType()!=FieldType.EMPTY) return false;
117         if(field[toX][toY].getPlayer()!=PlayerType.NONE) return false;//
118         
119         int xDif = toX - fromX;
120         int yDif = toY - fromY;
121         
122         if(Math.abs(xDif)>=0 && Math.abs(xDif)<=2 && Math.abs(yDif)>=0 && Math.abs(yDif)<=2) return true;
123         else return false;
124     }

126     public boolean isClonePossible(int fromX,int fromY,int toX,int toY){
127         if(!isMovePossible(fromX, fromY, toX, toY)) return false;
128         else{
129             int xDif = toX - fromX;
130             int yDif = toY - fromY;
131             
132             if(Math.abs(xDif)>=0 && Math.abs(xDif)<=1 && Math.abs(yDif)>=0 && Math.abs(yDif)<=1) return true;
133             else return false;
134         }
135     }
136     
137     public boolean isJumpPossible(int fromX,int fromY,int toX,int toY){
138         if(!isMovePossible(fromX, fromY, toX, toY)) return false;
139         else{
140             int xDif = toX - fromX;
141             int yDif = toY - fromY;
142             
143             if(Math.abs(xDif)>=0 && Math.abs(xDif)<=1 && Math.abs(yDif)>=0 && Math.abs(yDif)<=1) return false;
144             else return true;
145         }
146     }
147     
148     public boolean clone(int fromX, int fromY, int toX, int toY){
149         if(isClonePossible(fromX, fromY, toX, toY)){
150             field[toX][toY].setPlayer(field[fromX][fromY].getPlayer());
151             field[toX][toY].setType(FieldType.TAKEN);
152             ballCount.put(field[fromX][fromY].getPlayer(), ballCount.get(field[fromX][fromY].getPlayer())+1);
153             
154             for(int i=toX-1; i<=toX+1; i++){
155                 for(int j=toY-1; j<=toY+1; j++){
156                     if(i>=0 && i<width && j>=0 && j<height && !(i==toX && j==toY) && field[i][j].getFieldType() == FieldType.TAKEN){
157                         ballCount.put(field[i][j].getPlayer(), ballCount.get(field[i][j].getPlayer())-1);
158                         ballCount.put(field[fromX][fromY].getPlayer(), ballCount.get(field[fromX][fromY].getPlayer())+1);
159                         field[i][j].setPlayer(field[fromX][fromY].getPlayer());
160                     }
161                 }
162             }
163             
164             return true;
165         }
166         else return false;
167     }

169     public boolean jump(int fromX, int fromY, int toX, int toY){
170         if(isJumpPossible(fromX, fromY, toX, toY)){
171             field[toX][toY].setPlayer(field[fromX][fromY].getPlayer());
172             field[toX][toY].setType(FieldType.TAKEN);
173             
174             for(int i=toX-1; i<=toX+1; i++){
175                 for(int j=toY-1; j<=toY+1; j++){ 
176                     if(i>=0 && i<width && j>=0 && j<height && !(i==toX && j==toY) && field[i][j].getFieldType() == FieldType.TAKEN){
177                         ballCount.put(field[i][j].getPlayer(), ballCount.get(field[i][j].getPlayer())-1);
178                         ballCount.put(field[fromX][fromY].getPlayer(), ballCount.get(field[fromX][fromY].getPlayer())+1);
179                         field[i][j].setPlayer(field[fromX][fromY].getPlayer());
180                     }
181                 }
182             }
183             
184             field[fromX][fromY].setPlayer(PlayerType.NONE);
185             field[fromX][fromY].setType(FieldType.EMPTY);
186             
187             
188             return true;
189         }
190         else return false;
191     }

193     /**
194      * Returns true when selected player  is able to move anywhere. You can use this method to skip
195      * the turn if player can't move anywhere.
196      * @param playerType
197      * @return boolean
198      */
199     public boolean isPlayerMoveable(PlayerType playerType){
200         boolean moveable=false;
201         
202         for(int i=0;i<width;i++){
203             for(int j=0;j<height;j++){
204                 if(field[i][j].getPlayer()==playerType){
205                     for(int k=i-2;k<=i+2;k++){
206                         for(int l=j-2;l<=j+2;l++){
207                             if(k>=0 && l>=0 && k<width && l<height){
208                                 if(field[k][l].getFieldType()==FieldType.EMPTY){
209                                     moveable = true;
210                                     break;
211                                 }
212                             }
213                         }
214                     }
215                 }
216             }
217         }
218         
219         return moveable;
220     }
