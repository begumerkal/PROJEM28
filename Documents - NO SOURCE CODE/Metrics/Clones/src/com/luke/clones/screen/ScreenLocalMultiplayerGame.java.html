42  public class ScreenLocalMultiplayerGame implements Screen{
43      private StartScreen startScreen;
44      private ViewBoard viewBoard;
45      private Viewport mainViewport;
46      private Viewport layerViewport;
47      private Viewport messageViewport;
48      private Stage mainStage, layerStage, messageStage;
49      
50      public ScreenLocalMultiplayerGame(StartScreen startScreen){
51          this.startScreen=startScreen;
52          init();
53      }
54      
55      //startPlayer might be null (so he will be randomly selected)
56      public void newGame(Map map, PlayerMoveTimeLimitType playerMoveTimeLimitType){
57          viewBoard = new ViewBoard(map, playerMoveTimeLimitType, startScreen);
58          mainStage = viewBoard.getMainStage();
59          layerStage = viewBoard.getLayerStage();
60          messageStage = viewBoard.getMessageStage();
61          mainViewport = mainStage.getViewport();
62          layerViewport = layerStage.getViewport();
63          messageViewport = messageStage.getViewport();
64      }
65      
66      private void init(){
67          Label errorLabel = new Label("Game not started yet (use newGame())", new Skin(Gdx.files.internal("data/skins/skin_medium.json")));
68  //      mainStage = new Stage(Config.width,Config.height,false);
69          mainViewport = new FitViewport(Config.width, Config.height);
70          mainStage = new Stage();
71          mainStage.setViewport(mainViewport);
72          mainStage.addActor(errorLabel);
73      }
74      
75      @Override
76      public void show() {
77      }
78      
79      @Override
80      public void hide() {
81          mainStage = null;
82          layerStage = null;
83          messageStage=null;
84          viewBoard = null;
85      }
86      
87      @Override
88      public void render(float delta) {
89          Gdx.gl.glClearColor(54f/255f, 47f/255f, 45f/255f, 1);
90  //      Gdx.gl.glClear(GL10.GL_COLOR_BUFFER_BIT);
91          Gdx.gl.glClear(GL20.GL_COLOR_BUFFER_BIT);
92          
93          if(mainStage!=null){
94              mainStage.act();
95              mainStage.draw();
96          }
97          if(layerStage!=null){
98              layerStage.act();
99              layerStage.draw();
100         }
101         if(messageStage!=null){
102             messageStage.act();
103             messageStage.draw();
104         }
105         
106 //      if(Config.debug) Table.drawDebug(mainStage);
107     }
108 
109     @Override
110     public void resize(int width, int height) {
111         //Generalnie tutaj nale¿y resizowaæ te 3 viewporty z ró¿nych layer stagów
112         
113         if(mainViewport!=null) mainViewport.update(width, height, true);
114         if(layerViewport!=null) layerViewport.update(width, height, true);
115         if(messageViewport!=null) messageViewport.update(width, height, true);
116         
117 //      if(mainViewport!=null) mainViewport.update(Config.width, Config.height, true);
118     }
