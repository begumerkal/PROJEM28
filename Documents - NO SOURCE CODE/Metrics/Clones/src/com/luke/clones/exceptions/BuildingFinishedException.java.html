25  public class BuildingFinishedException extends RuntimeException{
26      private static final long serialVersionUID = -6117350005836843514L;
27      private boolean finished;
28      
29      public BuildingFinishedException(boolean finished){
30          super();
31          this.finished = finished;
32      }
33      
34      @Override
35      public String getLocalizedMessage() {
36          String msg="";
37          if(finished) msg = super.getLocalizedMessage() + "Can't make this kind of change when building is finished";
38          else if (!finished) msg = super.getLocalizedMessage() + "Finish building to use this function";
39          return msg;
40      }
41  
42      @Override
43      public String getMessage() {
44          String msg="";
45          if(finished) msg = super.getMessage() + "Can't make this kind of change when building is finished";
46          else if (!finished) msg = super.getMessage() + "Finish building to use this function";
47          return msg;
48      }
49  
50      @Override
51      public String toString() {
52          String msg="";
53          if(finished) msg = super.getLocalizedMessage() + "Can't make this kind of change when building is finished";
54          else if (!finished) msg = super.getLocalizedMessage() + "Finish building to use this function";
55          return super.toString() + msg;
56      } 
57          
58  }
