32  public class DatabaseService {
33      public final static DatabaseService INSTANCE = new DatabaseService();
34      
35      private Logger log = Logger.getLogger(DatabaseService.class);
36      private DatabaseWrapper databaseWrapper = new DatabaseWrapper();
37      
38      private DatabaseService(){
39  
40      }
41      
42      private void connect(){
43          try {
44              databaseWrapper.connect();
45              log.error("Successfully connected to database.");
46          } catch (SQLException e) {
47              log.error("Cannot connect to database. Stats cannot be saved.");
48              log.error("",e);
49          }
50      }
51      
52      private void connectToDatabaseIfNotConnected(){
53          try {
54              if(!databaseWrapper.isConnected()){
55                  log.error("There is no connection with database. Trying to connect...");
56                  connect();
57              }
58              else{
59  //              log.debug("There is connection with database already. Everything is fine. Stats are being saved.");
60              }
61          } catch (SQLException e) {
62              log.fatal("Fatal error occured. Cannot check if there is connection or not. " +
63                      "Probably there is no connection and stats cannot be saved.");
64              log.fatal("",e);
65          }
66      }
67      
68      private Connection getDatabaseConnection() throws SQLException {
69          if(databaseWrapper.getDatabaseConnection()==null || !databaseWrapper.isConnected()){
70              SQLException sqlException = new SQLException("No database connection - stats cannot be saved");
71              log.error("No database connection - stats cannot be saved");
72              log.error("",sqlException);
73              throw sqlException;
74          }
75          return databaseWrapper.getDatabaseConnection();
76      }
77      
78      public Connection connectToDatabaseIfNotConnectedAndReturnConnection() throws SQLException{
79          connectToDatabaseIfNotConnected();
80          return getDatabaseConnection();
81      }
82
