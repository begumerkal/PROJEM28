30  public class ButtonBar {
31      protected ImgButton[] imgButton;
32      private int x, y, buttonSpacing, yOffset;
33      private int buttonWidth, buttonHeight;
34      
35      public ButtonBar(Stage stage, Texture[] up, Texture[] down){
36          x=0;
37          y=0;
38          buttonWidth = 50;
39          buttonHeight = 50;
40          buttonSpacing = 0;
41          yOffset = 0;
42          
43          ButtonListener buttonListener = new ButtonListener();
44          imgButton = new ImgButton[up.length];
45          for(int i=0; i<up.length; i++){
46              imgButton[i] = new ImgButton(up[i],down[i]);
47              imgButton[i].setPosition(0, 0);
48              imgButton[i].setSize(50, 50);
49              imgButton[i].addListener(buttonListener);
50              stage.addActor(imgButton[i]);
51              imgButton[i].toFront();
52          }
53      }
54      
55      protected void click(InputEvent event, float x, float y){
56          
57      }
58      
59      public void removeFromStage(){
60          for(int i=0; i<imgButton.length; i++){
61              imgButton[i].remove();
62          }
63      }
64      
65      private class ButtonListener extends ClickListener{
66          @Override
67          public void clicked(InputEvent event, float x, float y) {
68              click(event, x, y);
69          }
70      }
71      
72      public void setPosition(int x, int y){
73          this.x = x;
74          this.y = y;
75          calculateSize();
76      }
77      
78      public void setButtonSize(int width, int height){
79          this.buttonWidth = width;
80          this.buttonHeight = height;
81          for(int i=0; i<imgButton.length; i++){
82              imgButton[i].setSize(width, height);
83          }
84          calculateSize();
85      }
86      
87      public void setSpacing(int buttonSpacing, int yOffset){
88          this.buttonSpacing = buttonSpacing;
89          this.yOffset = yOffset;
90          calculateSize();
91      }
92      
93      private void calculateSize(){
94          for(int i=0; i<imgButton.length; i++){
95              imgButton[i].setPosition(x + (imgButton[i].getWidth() + buttonSpacing)*i, y + yOffset);
96          }
97      }
98  }
