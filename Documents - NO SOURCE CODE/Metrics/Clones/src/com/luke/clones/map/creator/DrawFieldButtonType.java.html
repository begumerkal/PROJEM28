28  public enum DrawFieldButtonType {
29      SOLID(FieldType.SOLID, null),
30      TAKEN_RED(FieldType.TAKEN, Color.RED),
31      TAKEN_BLUE(FieldType.TAKEN, Color.BLUE),
32      TAKEN_GREEN(FieldType.TAKEN, Color.GREEN),
33      TAKEN_ORANGE(FieldType.TAKEN, Color.ORANGE);
34      
35      private FieldType drawFieldType = FieldType.TAKEN;
36      private Color drawFieldColor = Color.RED;
37      
38      private DrawFieldButtonType(FieldType drawFieldType, Color drawFieldColor) {
39          this.drawFieldType = drawFieldType;
40          this.drawFieldColor = drawFieldColor;
41      }
42  
43      public FieldType getDrawFieldType() {
44          return drawFieldType;
45      }
46  
47      public Color getDrawFieldColor() {
48          return drawFieldColor;
49      }
50      
51  }
